	on_opened_soundeffect = "event:/SFX/Events/europenorthamerica/before_the_battle"

	trigger = {
		# triggered by second_opium_war.4
	}

	immediate = {
		c:GBR = { save_scope_as = gbr_scope }
		c:RUS = { save_scope_as = rus_scope }
		s:STATE_OUTER_MANCHURIA = {
			random_scope_state = { save_scope_as = outer_manchuria_scope }
		}
	}

	option = {
		name = second_opium_war.5.a
		default_option = yes
		
		create_diplomatic_play = {
			war = yes
			type = dp_humiliation
			target_country = c:CHI
		}	
		every_diplomatic_play = {
			limit = {
				initiator = c:RUS
				target = c:CHI
			}
			add_war_goal = {
				holder = c:RUS
				type = return_state
				target_state = s:STATE_OUTER_MANCHURIA.region_state:HLJ
			}
			add_war_goal = {
				holder = c:RUS
				type = return_state
				target_state = s:STATE_OUTER_MANCHURIA.region_state:JLN
			}
			add_war_goal = {
				holder = c:RUS
				type = war_reparations
				target_country = c:CHI
			}
		}
		
		ai_chance = {
			base = 1
		}
	}
}
