#고부민란
donghak_events.7 = {
	type = country_event

	event_image = {
		texture = "gfx/event_pictures/bonggi.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	#on_opened_soundeffect = "event:/SFX/Events/asia/confucianism_shinto" 

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	title = donghak_events.7.t
	desc = donghak_events.7.d
	flavor = donghak_events.7.f

		trigger = {
		has_insurrectionary_interest_groups = yes
		ig:ig_rural_folk = {
			is_in_government = no
			is_marginal = no
			AND = {
				ig_approval > -8
			}
			NOT = { has_modifier = donghak_sangso_acceptance }
			NOT = { has_modifier = ig_mass_arrests }
			NOT = { has_variable = gobu_rebel }
		}
	}
	

	immediate = {
		set_variable = {
			name =  gobu_rebel
			value = yes
		}
		ig:ig_rural_folk = {
			save_scope_as = ig_rf_on_the_edge
			leader = {
				save_scope_as = union_leader
			}
			}
			ig:ig_landowners = {
			#limit = {
			#	is_in_government = yes
			#}
			leader = {
				save_scope_as = landowners_leader
			}
		}
	}

	option = { #진압한다
		default_option = yes
		name = donghak_events.7.a
		add_radicals = {
			pop_type = peasants
			culture = cu:korean
			value = 0.7
		}
		ig:ig_rural_folk = {
			join_revolution = yes
			add_modifier = {
				name = donghak_revolts
				months = 20
			}
			add_modifier = {
				name = donghak_gobu_banran
				months = 40
			}
		}
		scope:union_leader = {
			set_ideology = ideology_donghak
		}
	}

	option = {#요구를 받아들인다
		name = donghak_events.7.b
		scope:landowners_leader = {
				kill_character = yes
		}
		ig:ig_landowners = {
			add_modifier = {
				name = donghak_revolts_landowners
				months = 12
			}
		}
	}

}

	title = donghak_events.8.t
	desc = donghak_events.8.d
	flavor = donghak_events.8.f

	trigger = {
		has_modifier = donghak_gobu_banran
		NOT = { has_variable = baeksan }
	}
	
	immediate = {
		set_variable = {
			name = baeksan
			value = yes
		}
	}

option = {#동학농민운동
	name = donghak_events.8.a
		set_variable = {
			name = donghak_civil_war
			value = yes
		}
		create_country = {
			tag = ABC
			origin = this
			state = s:STATE_YANGHO.region_state:KOR
			on_created = { 
				add_modifier = {
					name = donghak_gun_conscription
					months = 24
				}
				activate_law = law_type:law_wage_subsidies
				activate_law = law_type:law_national_militia
				activate_law = law_type:law_regulatory_bodies
				activate_law = law_type:law_census_voting
				activate_law = law_type:law_freedom_of_conscience
				activate_law = law_type:law_elected_bureaucrats
				ig:ig_rural_folk = {
					add_ruling_interest_group = yes
				}
				ig:ig_devout = {
					add_ruling_interest_group = yes
				}
				if = {
					limit = {
						game_date < 1880.1.1
                                                game_date > 1875.1.1
					}
					create_character = {
						first_name = Bongjun
						last_name = Jeon
						historical = yes
						birth_date = 1855.1.10
						culture = cu:korean
						religion = rel:donghak
                                                dna = dna_jeonbongjun
						interest_group = ig_devout
						ideology = ideology_donghak 
						ruler = yes
						traits = {
							inspirational_orator
						}
						
					}
				}
				else = {
					create_character = {
						culture = cu:korean
						religion = rel:donghak
						interest_group = ig_devout
						ideology = ideology_donghak
						ruler = yes
					}
				}
			}
		}
	}
}
