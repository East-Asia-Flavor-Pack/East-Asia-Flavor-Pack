han_modifier_remove = {
	c:JAP = {
		every_scope_state = {
			limit = { 
				OR = {
					has_modifier = han_independency_very_high
					has_modifier = han_independency_high
					has_modifier = han_independency_medium
					has_modifier = han_independency_low
					has_modifier = han_independency_very_low
				}
			}
			remove_modifier = han_independency_very_high
			remove_modifier = han_independency_high
			remove_modifier = han_independency_medium
			remove_modifier = han_independency_low
			remove_modifier = han_independency_very_low
		}
	}
}
