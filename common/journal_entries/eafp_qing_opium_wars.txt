je_qing_opium_obsession = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_qing

	scripted_progress_bar = eafp_first_opium_war_debate_progress_bar

	immediate = {
		set_variable = { name = eafp_opium_policy_score value = 0 }
		set_variable = { name = eafp_opium_crisis_pressure value = 0 }
		trigger_event = { id = first_opium_war.1 popup = yes }
	}

	on_monthly_pulse = {
		random_events = {
			100 = 0
			40 = first_opium_war.2
			40 = first_opium_war.3
			30 = first_opium_war.4
			25 = first_opium_war.5
			10 = first_opium_war.6
			5 = first_opium_war.7
			5 = first_opium_war.8
			10 = first_opium_war.31
			10 = first_opium_war.32
			10 = first_opium_war.33
			10 = first_opium_war.34
			10 = first_opium_war.35
		}
		effect = {
			if = {
				limit = {
					OR = {
						je:je_qing_opium_obsession = {
							"scripted_bar_progress(eafp_first_opium_war_debate_progress_bar)" >= 100
						}
						var:eafp_opium_crisis_pressure >= 100
					}
					NOT = { has_variable = opium_wars_gave_in }
					NOT = { has_variable = opium_wars_target }
				}
				trigger_event = { id = first_opium_war.30 popup = yes }
			}
		}
	}

	complete = {
		OR = {
			has_variable = opium_wars_won
			has_variable = opium_wars_gave_in
			has_variable = opium_wars_target
			has_variable = lost_opium_wars
			has_modifier = opium_wars_lost
		}
	}

	invalid = {
		OR = {
			NOT = { exists = c:CHI }
			NOT = { THIS = c:CHI }
			NOT = { exists = c:GBR }
		}
	}

	on_timeout = {
		trigger_event = { id = first_opium_war.30 popup = yes }
	}

	weight = 10000
	should_be_pinned_by_default_uninvolved_or_context = yes

	timeout = 1460
}

je_qing_domestic_opium_substitution = {
	icon = "gfx/interface/icons/goods_icons/opium.dds"

	group = je_group_qing

	immediate = {
		set_variable = { name = eafp_domestic_opium_plantation_levels value = 0 }
	}

	on_monthly_pulse = {
		random_events = {
			100 = 0
			20 = first_opium_war.26
			20 = first_opium_war.27
		}
	}

	complete = {
		country_has_building_type_levels = {
			target = bt:building_opium_plantation
			value >= 20
		}
	}

	on_complete = {
		set_variable = eafp_domestic_opium_success
		trigger_event = { id = first_opium_war.29 popup = yes }
	}

	invalid = {
		OR = {
			NOT = { exists = c:CHI }
			NOT = { THIS = c:CHI }
			has_variable = opium_wars_target
			has_variable = opium_wars_gave_in
		}
	}

	on_timeout = {
		trigger_event = { id = first_opium_war.28 popup = yes }
	}

	timeout = 900
	weight = 1000
	should_be_pinned_by_default_uninvolved_or_context = yes
}
