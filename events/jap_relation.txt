#Duel
jap_relation.10 = { #The duel
	type = country_event
	placement = root

	title = jap_relation.10.t
	desc = jap_relation.10.d
	flavor = jap_relation.10.f

	duration = 3

	gui_window = event_window_2char

	left_icon = scope:duel_participant_1
	right_icon = scope:duel_participant_2

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/2Characters"

	icon = "gfx/interface/icons/event_icons/event_skull.dds"

	trigger = {
	}

	immediate = {
		random_country = {
			limit = { 
				NOT = {this=c:NIP}
				is_subject_type = subject_type_daimyo
				is_subject_of = c:JAP
			}
			save_scope_as = duel_country_1
			ruler = {
				save_scope_as = duel_participant_1
			}
		}

		random_country = {
			limit = { 
				NOT = {this=c:NIP}
				is_subject_type = subject_type_daimyo
				is_subject_of = c:JAP
				NOT = { this = scope:duel_country_1} 
			}
			save_scope_as = duel_country_2
			ruler = {
				save_scope_as = duel_participant_2
			}
		}
	}

	option = {#side to first daimyo
		default_option = yes
		name = jap_relation.10.a
		scope:duel_country_1 = {
			change_relations = {
					country = c:JAP
					value = 30
			}
			change_relations = {
					country = c:NIP
					value = -30
			}
		}
		scope:duel_country_2 = {
			change_relations = {
					country = c:JAP
					value = -30
			}
			change_relations = {
					country = c:NIP
					value = 30
			}
		}
	}

	option = {#side to second daimyo
		name = jap_relation.10.b
		scope:duel_country_2 = {
			change_relations = {
					country = c:JAP
					value = 30
			}
			change_relations = {
					country = c:NIP
					value = -30
			}
		}
		scope:duel_country_1 = {
			change_relations = {
					country = c:JAP
					value = -30
			}
			change_relations = {
					country = c:NIP
					value = 30
			}
		}
	}
	
}

#relief fund
jap_relation.20 = { 
	type = country_event
	placement = root

	title = jap_relation.20.t
	desc = jap_relation.20.d
	flavor = jap_relation.20.f

	gui_window = event_window_1char_tabloid

	left_icon = scope:relief_participant_1

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	icon = "gfx/interface/icons/event_icons/event_skull.dds"
	
	duration = 3

	trigger = {
	}

	immediate = {
		random_country = {
			limit = { 
				NOT = {this=c:NIP}
				is_subject_type = subject_type_daimyo
				is_subject_of = c:JAP
			}
			save_scope_as = relief_country_1
			ruler = {
				save_scope_as = relief_participant_1
			}
		}
	}

	option = {#fund him
		name = jap_relation.20.a
		default_option = yes
		ai_chance = {base = 9}

		add_modifier = {
			name = modifier_daimyo_relief
			multiplier = money_amount_multiplier_medium
			months = 60
		}

		scope:relief_country_1 = {
			add_modifier = {
				name = modifier_daimyo_relief_receive
				multiplier = money_amount_multiplier_medium
				months = 60
			}
			change_relations = {
					country = c:JAP
					value = 30
			}
			change_relations = {
					country = c:NIP
					value = -30
			}
		}
	}

	option = {#do not fund him

		name = jap_relation.20.b
		ai_chance = {base = 1}

		scope:relief_country_1 = {
			change_relations = {
					country = c:JAP
					value = -30
			}
			change_relations = {
					country = c:NIP
					value = 30
			}
		}
	}
}

	type = country_event
	placement = root

	desc = jap_relation.30.d
	flavor = jap_relation.30.f

	gui_window = event_window_1char_tabloid

	left_icon = scope:denno_participant_1

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"
	
	duration = 3

	trigger = {
	}

	immediate = {
		c:NIP={
			ruler = {
				save_scope_as = denno_participant_1
			}
		}
	}
#big celebration
	option = {
		default_option = yes
		ai_chance = {base = 9}
		name = jap_relation.30.a
		c:JAP = { 
			every_subject_or_below = {
				limit = {
					is_subject_type = subject_type_daimyo  
					NOT = {this=c:NIP}
				}
				change_relations = {
					country = c:NIP
					value = 5
				}
			}	
		}
	}
#small celebration
	option = {
		name = jap_relation.30.b
		ai_chance = {base = 1}
		c:JAP = { 
			every_subject_or_below = {
				limit = {
					is_subject_type = subject_type_daimyo  
					NOT = {this=c:NIP}
				}
				change_relations = {
					country = c:JAP
					value = -10
				}
			}	
		}
	}
}
