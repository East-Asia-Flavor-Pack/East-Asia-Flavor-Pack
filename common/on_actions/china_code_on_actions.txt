je_chinese_reunification_start_check = {
	effect = {
		every_scope_play_involved = {
			limit = {
				OR = {
					THIS = c:CHI
					country_has_primary_culture = cu:han
				}
				aggressive_diplomatic_plays_permitted = yes
				any_country = {
					is_diplomatic_play_enemy_of = PREV
					OR = {
						PREV = c:CHI
						country_has_primary_culture = cu:han
					}
				}
			}
			add_journal_entry = { type = je_chinese_reunification }
		}
	}
}

china_on_diplo_play_back_down = {
	effect = {
		if = {
			limit = {
				scope:actor = {
					any_scope_state = {
						state_is_in_china_proper = yes
						is_treaty_port = no
					}
				}
			}
			if = {
				limit = {
					initiator_is = scope:actor
					target = { 
						save_temporary_scope_as = concession_candidate_country_temp
						scope:actor = {
							any_scope_state = {
								state_is_in_china_proper = yes
								is_treaty_port = no
								NOT = { is_target_in_variable_list = { name = concession_owner_varlist target = scope:concession_candidate_country_temp } }
							}
						}
					}
				}
				save_scope_value_as = { name = concession_gain_cause value = 1 } # 무력으로
				save_scope_value_as = { name = concession_gain_country_num value = 1 }
				target = { save_scope_as = concession_candidate_country }
				scope:actor = {
					save_scope_as = concession_target_country
					trigger_event = concessions_in_china.1002
				}
				target = { trigger_event = concessions_in_china.1001 }
			}
			else_if = {
				limit = {
					target_is = scope:actor
					initiator = { 
						save_temporary_scope_as = concession_candidate_country_temp
						scope:actor = {
							any_scope_state = {
								state_is_in_china_proper = yes
								is_treaty_port = no
								NOT = { is_target_in_variable_list = { name = concession_owner_varlist target = scope:concession_candidate_country_temp } }
							}
						}
					}
				}
				save_scope_value_as = { name = concession_gain_cause value = 1 } # 무력으로
				save_scope_value_as = { name = concession_gain_country_num value = 1 }
				initiator = { save_scope_as = concession_candidate_country }
				scope:actor = {
					save_scope_as = concession_target_country
					trigger_event = concessions_in_china.1002
				}
				initiator = { trigger_event = concessions_in_china.1001 }
			}
		}
	}
}

china_on_self_capitulated_notification = {
	effect = {
		if = {
			limit = {
				any_scope_state = {
					state_is_in_china_proper = yes
					is_treaty_port = no
				}
			}
			set_variable = {
				name = concession_gain_country_num
				value = 0
			}
			save_scope_as = concession_target_country
			save_scope_value_as = { name = concession_gain_cause value = 1 } # 무력으로
			every_country = {
				limit = {
					has_variable = concession_candidate_country
					save_temporary_scope_as = concession_candidate_country_temp
					scope:concession_target_country = {
						any_scope_state = {
							state_is_in_china_proper = yes
							is_treaty_port = no
							NOT = { is_target_in_variable_list = { name = concession_owner_varlist target = scope:concession_candidate_country_temp } }
						}
					}
				}
				scope:concession_target_country = {
					change_variable = {
						name = concession_gain_country_num
						add = 1
					}
				}
				trigger_event = concessions_in_china.1001
			}
			every_country = {
				limit = { has_variable = concession_candidate_country }
				remove_variable = concession_candidate_country
			}
			if = {
				limit = { var:concession_gain_country_num > 0 }
				save_scope_value_as = {
					name = concession_gain_country_num
					value = var:concession_gain_country_num
				}
				trigger_event = concessions_in_china.1002
			}
			remove_variable = concession_gain_country_num
		}
		else = {
			every_country = {
				limit = { has_variable = concession_candidate_country }
				remove_variable = concession_candidate_country
			}
		}
	}
}
