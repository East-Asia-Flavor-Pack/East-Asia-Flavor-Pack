jpf_joseon.1 = { # 기해박해 시작 : 우의정이 아룀
	type = country_event
	placement = ROOT

	title = jpf_joseon.1.t
	desc = jpf_joseon.1.d
	flavor = jpf_joseon.1.f

	event_image = {
		video = "gfx/event_pictures/asia_union_leader.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/asia/union_leader"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	duration = 3

	trigger = {
		# je_jpf_vicious_cult 저널이 트리거함
	}

	immediate = {
	}
	option = { # 박해 시작
		name = jpf_joseon.1.a
		default_option = yes
		custom_tooltip = jpf_joseon.1.a.tt
		hidden_effect = {
			trigger_event = { id = jpf_joseon.2 days = 30}
		}
	}
	option = { # 박해 안함
		name = jpf_joseon.1.b
		default_option = yes
		custom_tooltip = jpf_joseon.1.b.tt
		every_interest_group = {
			limit = {
				law_stance = {
					law = law_type:law_state_religion
					value > neutral
				}
			}
			add_modifier = {
				name = jpf_ignored
				months = short_modifier_time
			}
		}
		every_interest_group = {
			limit = {
				law_stance = {
					law = law_type:law_state_religion
					value < neutral
				}
			}
			add_modifier = {
				name = jpf_repected
				months = short_modifier_time
			}
		}
	}
}
