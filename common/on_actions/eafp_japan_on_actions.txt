
eafp_japan_on_monthly_pulse_country = {
	effect = {
		if = {
			limit = {
                has_variable = sonnoha_var_first
				var:sonnoha_var >= 150
			}
			remove_variable = sonnoha_var_first
			set_variable = sonnoha_var_second
			change_variable = {
				name = sonnoha_var
				subtract = 150
			}
		}
		if = {
			limit = {
                has_variable = sonnoha_var_second
				var:sonnoha_var >= 150
			}
			remove_variable = sonnoha_var_second
			set_variable = sonnoha_var_third
			change_variable = {
				name = sonnoha_var
				subtract = 150
			}
		}
	}
}

eafp_japan_on_half_yearly_pulse_country = {
	effect = {
		if = {
			limit = { NOT = { has_variable = boshin_war_var }}
			han_modifier_kyushu = yes
			han_modifier_tohoku = yes
			han_modifier_shikoku = yes
			han_modifier_chugoku = yes
			han_modifier_chubu = yes
			han_modifier_kansai = yes
		}
		if = {
			limit = { has_variable = boshin_war_var }
			han_modifier_remove = yes
		}
	}
}
