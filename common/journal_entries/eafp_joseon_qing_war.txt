		has_war_with = c:CHI

				NOT = { has_war_with = c:CHI }
				NOT = { is_subject_of = c:CHI }

			name = eafp_var_jqw_result_grade

			set_variable = {
				name = eafp_var_jqw_result_grade
				value = 4
			}
		}

			limit = {
				scope:journal_entry = {
					"scripted_bar_progress(eafp_joseon_qing_war_evaluation_progress_bar)" >= 65

		remove_modifier = eafp_modifier_joseon_qing_war_holding

		remove_modifier = eafp_modifier_joseon_qing_war_exhaustion

	}

		NOT = { has_war_with = c:CHI }
		is_subject_of = c:CHI
	}

	on_fail = {
		set_variable = {
			name = eafp_var_jqw_result_grade
			value = 1
		}
		if = {
			limit = {
				scope:journal_entry = {
					"scripted_bar_progress(eafp_joseon_qing_war_evaluation_progress_bar)" >= 70

				name = eafp_var_jqw_result_grade
				value = 4
			}

			limit = {
				scope:journal_entry = {
					"scripted_bar_progress(eafp_joseon_qing_war_evaluation_progress_bar)" >= 45

				var:eafp_var_joseon_qing_war_months <= 36
			}

				name = eafp_var_jqw_result_grade
				value = 3

			limit = {
				scope:journal_entry = {
					"scripted_bar_progress(eafp_joseon_qing_war_evaluation_progress_bar)" >= 20

				var:eafp_var_joseon_qing_war_months <= 36

				name = eafp_var_jqw_result_grade

	}

		exists = c:CHI
		NOT = { exists = scope:qing_enemy }
		NOT = { has_war_with = c:CHI }
	}

		remove_variable = eafp_var_jqw_bad_shells_coverup
		remove_variable = eafp_var_jqw_transport_reformed
		remove_modifier = eafp_modifier_joseon_qing_war_momentum
		remove_modifier = eafp_modifier_joseon_qing_war_holding
		remove_modifier = eafp_modifier_joseon_qing_war_strained_effort

					name = eafp_modifier_joseon_qing_war_momentum
				}

			else_if = {
				limit = {
					scope:journal_entry = {
						"scripted_bar_progress(eafp_joseon_qing_war_evaluation_progress_bar)" >= 55
					}

				add_modifier = {
					name = eafp_modifier_joseon_qing_war_holding
				}
			}

				limit = {
					scope:journal_entry = {
						"scripted_bar_progress(eafp_joseon_qing_war_evaluation_progress_bar)" >= 30

					name = eafp_modifier_joseon_qing_war_strained_effort
				}
			}
