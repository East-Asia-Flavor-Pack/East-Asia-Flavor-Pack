daimyo = {
	groups= {
		general # for breaking only
	}
	show_in_lens = no
	can_use_obligations = yes

	potential = {
		always = no
	}

	possible = {
		always = no	
	}


	pact = {	
		cost = 100
		has_junior_participant = yes
		recipient_pays_maintenance = no

		income_transfer = 0.1
		second_country_gets_income_transfer = no
		income_transfer_based_on_second_country = yes
		
		relations_progress_per_day = 0
		relations_improvement_max = 50

		target_requires_approval_to_break = yes	
		
		subject_type = subject_type_daimyo
		
		second_modifier = {
			country_convoy_contribution_to_market_owner_add = 0.25
		}

		actor_can_break = {
			always = no
		}

		target_can_break = {
			always = no
		}
		
		manual_break_effect = {
			create_bidirectiona_truce = { country = scope:target_country months = 60 }
		}
		
		auto_break_effect = {
			create_bidirectional_truce = { country = scope:target_country months = 60 }
		}
	}
	
	accept_effect = {
		create_incident = {
			country = scope:target_country
			value = 5
		}
	}
	
	ai = {
		evaluation_chance = {
			value = 0
		}

		will_break = {
			always = no		
		}
	}
}
