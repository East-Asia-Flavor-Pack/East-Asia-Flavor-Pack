
je_korean_empire = {

 icon = "gfx/interface/icons/event_icons/event_scales.dds"

	immediate = {
		set_variable = korean_empire_journal
	}


	complete = {
		ROOT = {
			country_rank >= rank_value:major_power
		}
		is_subject = no
	}

	
	on_complete = {
		trigger_event = {id = korean_empire.1 }
	}

	fail = {
		custom_tooltip = {
			text = joseon_yuji
			has_variable = reformation_of_qing
		}
	}

	on_fail = {
		hidden_effect = {
		trigger_event = { id = reformation_of_qing.1 days = 90 }
		}
	}

	weight = 10000

	should_be_pinned_by_default = yes
}

je_hikikomori = {

	icon = "gfx/interface/icons/event_icons/event_scales.dds"


	immediate = {
		add_modifier = {
			name = hikikomori_md
		}
	}


	 complete = {
		any_scope_state = {
			filter = {
				is_incorporated = yes
			}
			percent >= 0.75
			any_scope_building = {
				is_building_type = building_urban_center
				level >= 10
			}
		}
		any_scope_state = {
			is_incorporated = yes
			any_scope_building = {
				is_building_type = building_urban_center
				level >= 30
			}
		}
	}

   on_complete = {
		trigger_event = { id = hikikomori.1 }
	}

}
