je_eafp_joseon_qing_war = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_crises

	scripted_progress_bar = eafp_joseon_qing_war_evaluation_progress_bar

	is_shown_when_inactive = {
		c:KOR ?= THIS
		exists = c:CHI
		has_war_with = c:CHI
	}

	possible = {
		c:KOR ?= THIS
		exists = c:CHI
		has_war_with = c:CHI
	}

	immediate = {
		c:CHI = {
			save_scope_as = qing_enemy
		}
		remove_variable = eafp_var_joseon_qing_war_months
		remove_variable = eafp_var_jqw_result
		remove_variable = eafp_var_jqw_result_grade
		remove_variable = eafp_var_jqw_kor_enforced_goal
		remove_variable = eafp_var_jqw_qing_enforced_goal
		remove_variable = eafp_var_jqw_kor_capitulated
		remove_variable = eafp_var_jqw_bad_shells_coverup
		remove_variable = eafp_var_jqw_transport_reformed
		set_variable = {
			name = eafp_var_joseon_qing_war_months
			value = 0
		}
		trigger_event = { id = eafp_joseon_qing_war_events.1 popup = yes }
	}

	complete = {
		OR = {
			NOT = { exists = c:CHI }
			AND = {
				exists = c:CHI
				NOT = { has_war_with = c:CHI }
				NOT = { is_subject_of = c:CHI }
				NOT = { has_variable = eafp_var_jqw_kor_capitulated }
				NOT = {
					AND = {
						has_variable = eafp_var_jqw_qing_enforced_goal
						NOT = { has_variable = eafp_var_jqw_kor_enforced_goal }
					}
				}
			}
		}
	}

	on_complete = {
		set_variable = {
			name = eafp_var_jqw_result
			value = 2
		}
		if = {
			limit = {
				NOT = { exists = c:CHI }
			}
			set_variable = {
				name = eafp_var_jqw_result
				value = 3
			}
		}
		else_if = {
			limit = {
				has_variable = eafp_var_jqw_kor_enforced_goal
				has_variable = eafp_var_jqw_qing_enforced_goal
			}
			set_variable = {
				name = eafp_var_jqw_result
				value = 1
			}
		}
		else_if = {
			limit = {
				has_variable = eafp_var_jqw_kor_enforced_goal
			}
			set_variable = {
				name = eafp_var_jqw_result
				value = 3
			}
		}

		set_variable = {
			name = eafp_var_jqw_result_grade
			value = 1
		}
		if = {
			limit = {
				scope:journal_entry = {
					"scripted_bar_progress(eafp_joseon_qing_war_evaluation_progress_bar)" >= 67
				}
			}
			set_variable = {
				name = eafp_var_jqw_result_grade
				value = 3
			}
		}
		else_if = {
			limit = {
				scope:journal_entry = {
					"scripted_bar_progress(eafp_joseon_qing_war_evaluation_progress_bar)" >= 34
				}
			}
			set_variable = {
				name = eafp_var_jqw_result_grade
				value = 2
			}
		}
		trigger_event = { id = eafp_joseon_qing_war_events.2 popup = yes }
	}

	fail = {
		exists = c:CHI
		NOT = { has_war_with = c:CHI }
		OR = {
			is_subject_of = c:CHI
			has_variable = eafp_var_jqw_kor_capitulated
			AND = {
				has_variable = eafp_var_jqw_qing_enforced_goal
				NOT = { has_variable = eafp_var_jqw_kor_enforced_goal }
			}
		}
	}

	on_fail = {
		set_variable = {
			name = eafp_var_jqw_result
			value = 0
		}
		set_variable = {
			name = eafp_var_jqw_result_grade
			value = 1
		}
		if = {
			limit = {
				scope:journal_entry = {
					"scripted_bar_progress(eafp_joseon_qing_war_evaluation_progress_bar)" >= 67
				}
			}
			set_variable = {
				name = eafp_var_jqw_result_grade
				value = 3
			}
		}
		else_if = {
			limit = {
				scope:journal_entry = {
					"scripted_bar_progress(eafp_joseon_qing_war_evaluation_progress_bar)" >= 34
				}
			}
			set_variable = {
				name = eafp_var_jqw_result_grade
				value = 2
			}
		}
		trigger_event = { id = eafp_joseon_qing_war_events.3 popup = yes }
	}

	invalid = {
		NOT = { c:KOR ?= THIS }
	}

	on_invalid = {
		remove_variable = eafp_var_joseon_qing_war_months
		remove_variable = eafp_var_jqw_result
		remove_variable = eafp_var_jqw_result_grade
		remove_variable = eafp_var_jqw_kor_enforced_goal
		remove_variable = eafp_var_jqw_qing_enforced_goal
		remove_variable = eafp_var_jqw_kor_capitulated
		remove_variable = eafp_var_jqw_bad_shells_coverup
		remove_variable = eafp_var_jqw_transport_reformed
	}

	on_monthly_pulse = {
		effect = {
			change_variable = {
				name = eafp_var_joseon_qing_war_months
				add = 1
			}
		}
		random_events = {
			85 = 0
			5 = eafp_joseon_qing_war_events.10
			4 = eafp_joseon_qing_war_events.11
			3 = eafp_joseon_qing_war_events.12
			2 = eafp_joseon_qing_war_events.13
			4 = eafp_joseon_qing_war_events.14
		}
	}

	weight = 7000
	can_deactivate = no

	should_be_pinned_by_default_uninvolved_or_context = yes
}
