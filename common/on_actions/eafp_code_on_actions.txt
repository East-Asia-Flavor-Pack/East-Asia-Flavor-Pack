eafp_on_diplo_play_back_down = {
	trigger = {
		scope:actor ?= {
			has_journal_entry = je_eastasian_expansionism
		}
	}
	effect = {
		scope:actor ?= {
			je:je_eastasian_expansionism = {
				add_progress = {
					name = eastasian_expansionism_paranoia_progress_bar
					value = 20
				}
			}
		}
	}
}

eafp_on_wargoal_enforced = {
	effect = {
		if = {
			limit = {
				scope:target ?= {
					has_journal_entry = je_eastasian_expansionism
				}
			}
			scope:target ?= {
				set_variable = { 
					name = temp_wargoal_impact 
					value = {
						value = scope:wargoal_impact 
						multiply = 1
					}
				}
				je:je_eastasian_expansionism = {
					add_progress = {
						value = owner.var:temp_wargoal_impact
						name = eastasian_expansionism_paranoia_progress_bar
					}
				}
			}
		}
		if = {
			limit = {
				owner = {
					has_journal_entry = je_eastasian_expansionism
				}
			}
			owner = {
				set_variable = { 
					name = temp_wargoal_impact 
					value = {
						value = scope:wargoal_impact 
						multiply = -1
					}
				}
				je:je_eastasian_expansionism = {
					add_progress = {
						value = owner.var:temp_wargoal_impact
						name = eastasian_expansionism_paranoia_progress_bar
					}
				}
			}
		}
	}
}

eafp_on_chinese_interregnum_wargoal_enforced = {
	effect = {
		if = {
			limit = {
				owner ?= {
					has_variable = eafp_claiming_chinese_central_government
					eafp_is_chinese_fragment = yes
				}
				scope:target ?= {
					eafp_is_chinese_central_government = yes
				}
			}
			owner = {
				remove_variable = eafp_claiming_chinese_central_government
				set_variable = chinese_central_government
				set_global_variable = {
					name = chinese_central_government
					value = THIS
				}
				eafp_create_chinese_interregnum_bloc_effect = yes
			}
			scope:target ?= {
				remove_variable = chinese_central_government
			}
			eafp_add_chinese_fragments_to_interregnum_effect = yes
		}
	}
}
