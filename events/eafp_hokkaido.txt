hokkaido.3 = { # 축성 부지 확보
	type = country_event
	placement = s:STATE_HOKKAIDO.region_state:JAP

	title = hokkaido.3.t
	desc = hokkaido.3.d
	flavor = hokkaido.3.f

	icon = "gfx/interface/icons/event_icons/tutorial_icon.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	 event_image = { 
		video = "asia_confucianism_shinto"
	}
	
	duration = 3
	
	immediate = {}

	trigger = {}
	
	option = { # 비용 대기
		default_option = yes
		name = hokkaido.3.a
		if = {
			limit = { scope:hokkaido2 = 1 }
			s:STATE_HOKKAIDO.region_state:JAP = {
				add_modifier = {
					name = fukuyama_castle_construction_han_modifier
					months = normal_modifier_time
				}
			}
			add_modifier = {
				name = fukuyama_castle_construction_bakufu_modifier
				months = normal_modifier_time
			}
		}
		else = {
			s:STATE_HOKKAIDO.region_state:JAP = {
				add_modifier = {
					name = fukuyama_castle_construction_han_modifier
					months = normal_modifier_time
					multiplier = 2
				}
			}
			add_modifier = {
				name = fukuyama_castle_construction_bakufu_modifier
				months = normal_modifier_time
				multiplier = 2
			}
		}
		trigger_event = { id = hokkaido.3 }
	}

	option = { # 알아서 하라 해
		default_option = yes
		name = hokkaido.3.b
		if = {
			limit = { scope:hokkaido2 = 1 }
			s:STATE_HOKKAIDO.region_state:JAP = {
				add_modifier = {
					name = fukuyama_castle_construction_han_modifier
					months = normal_modifier_time
					multiplier = 2
				}
			}
		}
		else = {
			s:STATE_HOKKAIDO.region_state:JAP = {
				add_modifier = {
					name = fukuyama_castle_construction_han_modifier
					months = normal_modifier_time
					multiplier = 4
				}
			}
			if = {
				limit = { has_journal_entry = je_bakuhantaisei_TOHOKU }
				custom_tooltip = {
					text = hokkaido.3.ttb2
					je:je_bakuhantaisei_TOHOKU = {
						add_progress = {
							name = bakuhantaisei_TOHOKU_loyalty_progress_bar
							value = -10
						}
					}
				}
			}
		}
	}

	after = {
		trigger_event = {
			id = hokkaido.4
			months = normal_modifier_time
		}
	}
}
