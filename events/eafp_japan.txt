eafp_japan.1001 = { #홋카이도 개척 완료
    type = country_event
    placement = root

    title = eafp_japan.1001.t
    desc = eafp_japan.1001.d
    flavor = eafp_japan.1001.f

    icon = "gfx/interface/icons/event_icons/tutorial_icon.dds"

    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

     event_image = { 
        video = "asia_confucianism_shinto"
	}
    
    duration = 3
    
    immediate = {}

    trigger = {}
    
    option = {
        default_option = yes
        name = eafp_japan.1001.a
        custom_tooltip = {
			text = eafp_japan.1001.a_tt
			
			every_scope_state = {
				limit = {
					NOT = {
						s:STATE_HOKKAIDO.region_state:JAP = THIS
					}
				}

				every_scope_pop = {
					limit = {
						has_pop_culture = japanese
						is_pop_type = peasants
					}
					change_pop_culture = { target = cu:korean value = 0.001 }
				}
				every_scope_pop = {
					limit = {
						has_pop_culture = korean
					}
					move_pop = s:STATE_HOKKAIDO.region_state:JAP
				}
			}
			s:STATE_HOKKAIDO.region_state:JAP = {
				every_scope_pop = {
					limit = {
						has_pop_culture = korean
					}
					change_pop_culture = { target = cu:japanese value = 1.0 }
				}
			}
		}
    }

    option = {
		name = eafp_japan.1001.b

		custom_tooltip = {
			text = eafp_japan.1001.b_tt
			
			s:STATE_KANTO.region_state:JAP = {

				every_scope_pop = {
					limit = {
						has_pop_culture = japanese
						is_pop_type = peasants
					}
					change_pop_culture = { target = cu:korean value = 0.025 }
				}
				every_scope_pop = {
					limit = {
						has_pop_culture = korean
					}
					move_pop = s:STATE_HOKKAIDO.region_state:JAP
				}
			}
			s:STATE_HOKKAIDO.region_state:JAP = {
				every_scope_pop = {
					limit = {
						has_pop_culture = korean
					}
					change_pop_culture = { target = cu:japanese value = 1.0 }
				}
			}
		}
		ai_chance = {
			base = 0
		}
	}
}



eafp_japan.2000 = { #텐포 대기근
    type = country_event
    placement = root

    title = eafp_japan.2000.t
    desc = eafp_japan.2000.d
    flavor = eafp_japan.2000.f

    icon = "gfx/interface/icons/event_icons/tutorial_icon.dds"

    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

     event_image = { 
        video = "asia_confucianism_shinto"
	}

    duration = 3
    
    immediate = {
        random_scope_state = {
            limit = {
				any_scope_building = {
					is_plantation_building = yes
				}
			}
			save_scope_as = tenpo_state
			random_scope_building = {
				limit = {
					is_plantation_building = yes
				}
				save_scope_as = tenpo_building
			}
        }
    }

    trigger = {}
    
    option = { # 해결 노력
		name = eafp_japan.2000.a
		default_option = yes
        add_journal_entry = {
            type = japan_bakufu_kaikaku_je
        }
		add_modifier = {
			name = modifier_crop_failure_relief
			multiplier = money_amount_multiplier_small
			months = short_modifier_time
		}
		scope:tenpo_state = {
			add_modifier = {
				name = modifier_crop_failure_tenpo
				months = normal_modifier_time
			}
			add_modifier = {
				name = nomin_genzei
				months = normal_modifier_time
			}
		}
		ai_chance = {
			base = 50
		}
	}

	option = { #해결 포기
		name = eafp_japan.2000.b
        add_journal_entry = {
            type = japan_bakufu_kaikaku_je
        }
		scope:tenpo_state = {
			add_modifier = {
				name = modifier_crop_failure_tenpo
				months = normal_modifier_time
			}
		}
		custom_tooltip = natural_disaster_variable_tt
		hidden_effect = {
			random_list = {
				50 = {
					random_scope_state = {
						limit = {
							is_incorporated = yes
						}
						set_variable = {
							name = natural_disaster_variable
							months = short_modifier_time
						}
					}
				}
				50 = {
				}
			}
		}
	}
}
