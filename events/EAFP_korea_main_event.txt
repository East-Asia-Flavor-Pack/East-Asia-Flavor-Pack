korea_main.1000 = { #실학 이벤트 체인 시작
    type = country_event
    placement = root

    duration = 3

    is_popup = yes

    title = korea_main.1000.t
    desc = korea_main.1000.d
    flavor = korea_main.1000.f

    icon = "gfx/interface/icons/ig_icons/intelligensia.dds"

    gui_window = event_window_1char_tabloid

	left_icon = scope:jeongyakyong_scope

    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
    on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

    immediate = {
        create_character = {
			first_name = yakyong
			last_name = jeong
			historical = yes
			birth_date = 1762.8.5
			interest_group = ig_industrialists
			traits = {
				innovative
				charismatic
			}
            save_scope_as = jeongyakyong_scope
		}
    }

    option = {
        name = korea_main.1000.a
        default_option = yes
        add_journal_entry = {
			type = je_korea_reformation_start
		}
        scope:jeongyakyong_scope = {
            kill_character = yes
        }
        if = {
            limit = {
                ig:ig_intelligentsia = {
                    is_being_bolstered = no
                }
            }
            ig:ig_intelligentsia = { #지식인 후원 ON
                set_ig_bolstering = yes
            }
        }
        custom_tooltip = { #실학파 영향력 증가
            text = je_korea_reformation_start_option_tt_positive
            change_variable = {
                name = silhak_power
                add = 5
            }
        }
        custom_tooltip = { # 왕권 감소
            text = royal_authority_subtract_five
            change_variable = {
                name = royal_authority
                subtract = 5
            }
        }
    }

    option = {
        name = korea_main.1000.b
        default_option = yes
        add_journal_entry = {
			type = je_korea_reformation_start
		}
        scope:jeongyakyong_scope = {
            kill_character = yes
        }
        custom_tooltip = { # 왕권 증가
            text = royal_authority_add_five
            change_variable = {
                name = royal_authority
                add = 5
            }
        }
    }

    option = {
        name = korea_main.1000.c
        default_option = yes
        add_journal_entry = {
			type = je_korea_reformation_start
		}
        scope:jeongyakyong_scope = {
            kill_character = yes
        }
        if = {
            limit = {
                ig:ig_intelligentsia = {
                    is_being_suppressed = no
                }
            }
            ig:ig_intelligentsia = { # 지식인 탄압 ON
                set_ig_suppression = yes
            }
        }
        custom_tooltip = { # 실학파 영향력 감소
            text = je_korea_reformation_start_option_tt_negative
            change_variable = {
                name = silhak_power
                subtract = 10
            }
        }
        custom_tooltip = { # 왕권 증가
            text = royal_authority_add_ten
            change_variable = {
                name = royal_authority
                add = 10
            }
        }
    }
}
