je_eafp_manchurian_policy = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_foreign_affairs

	is_shown_in_lobby = {
		c:KOR ?= this
	}

	is_shown_when_inactive = {
		has_variable = eafp_var_sam_all_clear
	}

	possible = {
		custom_tooltip = {
			text = has_completed_korean_reformation_military_je_tt
			has_variable = completed_je_korean_reformation_military
		}
		OR = {
			custom_tooltip = {
				text = has_completed_korean_reformation_society_je_tt
				has_variable = completed_je_korean_reformation_society
			}
			custom_tooltip = {
				text = has_completed_korean_reformation_economy_je_tt
				has_variable = completed_je_korean_reformation_economy
			}
			custom_tooltip = {
				text = has_completed_korean_reformation_diplomacy_je_tt
				has_variable = completed_je_korean_reformation_diplomacy
			}
		}
	}

	immediate = {
		trigger_event = { id = eafp_manchuria.1 popup = yes }
	}

	complete = {
		any_state_in_dongbei = {
			count = all
			owner = ROOT
		}
	}

	on_complete = {
		trigger_event = { id = eafp_manchuria.2 popup = yes }
	}

	event_outcome_completed_effect_desc = {
		header = eafp_manchuria.2.a
		effect = {
			add_journal_entry = {
				type = je_eafp_manchurian_management
			}
		}
	}

	weight = 6000
	can_deactivate = no

	should_be_pinned_by_default_uninvolved_or_context = yes
}

je_eafp_manchurian_management = {
	icon = "gfx/interface/icons/event_icons/event_trade.dds"

	group = je_group_foreign_affairs

	is_shown_when_inactive = {
		OR = {
			has_journal_entry = je_eafp_manchurian_policy
			any_state_in_dongbei = {
				count = all
				owner = ROOT
			}
		}
	}

	scripted_button = button_je_eafp_manchuria_settlement
	scripted_button = button_je_eafp_manchuria_expand_arable_land

	possible = {
		any_state_in_dongbei = {
			count = all
			owner = ROOT
		}
	}

	immediate = {
		
	}

	complete = {
		any_state_in_dongbei = {
			count = all
			owner = ROOT
			total_urbanization >= 1000
			culture_percent_state = {
				target = cu:korean
				value >= 0.10
			}
			tax_capacity >= tax_capacity_usage
		}
	}

	on_complete = {
		trigger_event = { id = eafp_manchuria.3 popup = yes }
	}

	event_outcome_completed_effect_desc = {
		header = eafp_manchuria.3.a
		effect = {
			every_scope_state_in_dongbei = {
				set_state_type = incorporated
				add_modifier = {
					name = eafp_manchurian_integrated_state
					days = long_modifier_time
				}
				add_cultural_community = cu:korean
				add_loyalists_in_state = {
					value = 0.1
				}
			}
		}
	}

	weight = 6000

	should_be_pinned_by_default_uninvolved_or_context = yes
}
