je_heavenly_kingdom_main = {
	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	is_shown_when_inactive = {
		exists = c:CHI # 추가
		THIS = c:CHI
	}

	possible = {
		is_subject = no
		any_scope_state = {
			is_incorporated = yes
			any_scope_pop = {
				has_pop_religion = protestant
			}
			state_region = { is_homeland = cu:han } # 추가
			is_capital = no
			turmoil > 0.15
		}
	}

	immediate = {
		random_scope_state = {
			limit = {
				is_incorporated = yes
				any_scope_pop = {
					has_pop_religion = protestant
				}
				NOT = { is_capital = yes }
				state_region = { is_homeland = cu:han } # 추가
			}
			add_modifier = {
				name = god_worshippers
				months = 43	#3.5년
			}
		}
		trigger_event = {
			id = heavenly.2
		}
	}

	on_monthly_pulse = {
		random_events = {
			5 = 0
			1 = heavenly.3
		}
	}

	complete = {
		NOT = {
			any_scope_state = {
				has_modifier = god_worshippers
			}
		}
	}

	fail = {
		any_scope_state = {
			has_modifier = god_worshippers
			count >= 4
		}
		any_scope_state = {
			has_modifier = god_worshippers
			state_region = { is_homeland = cu:han } # 추가
			any_scope_pop = {
				has_pop_religion = protestant
				religion_accepted = no
			}
			turmoil >= 0.25 # 수정
		}
	}

	on_fail = {
		trigger_event = {
			id = heavenly.4
		}
	}

	weight = 10000
	should_be_pinned_by_default = yes
}

je_taiping = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	complete = {
		custom_tooltip = {
			text = taiping_destroyed
			hidden_trigger = {
				NOT = {
					exists = c:TPG
				}
			}
		}
	}

	on_complete = {
		trigger_event = { id = heavenly.5 days = 0 }
	}

	fail = {
		exists = c:TPG
		has_truce_with = c:TPG
	}

	on_fail = {
		if = {
			limit = { has_journal_entry = je_warlord_china }
			post_notification = warlord_china_explosion_point_add
			change_variable = {
				name = china_warlord_explosion
				add = 1
			}
		}
		hidden_trigger = {
			random_list = {
				1 = {
					add_journal_entry = { type = je_taiping_truce_chi_attack }
					c:TPG = { add_journal_entry = { type = je_taiping_truce_tpg_defence } }
				}
				1 = {
					add_journal_entry = { type = je_taiping_truce_chi_defence }
					c:TPG = { add_journal_entry = { type = je_taiping_truce_tpg_attack } }
				}
			}
		}
	}

	weight = 1000
	should_be_pinned_by_default = yes

	inheritable = no
}
