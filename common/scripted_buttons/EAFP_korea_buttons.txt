kor_tutorial_button = {
	name = "kor_tutorial_button"
	desc = "kor_tutorial_button_desc"

	visible = {
		always = yes
	}

	possible = {
		root.var:kor_tutorial_var < 30
	}

	effect = {
		trigger_event = {
			id = korea_main.1
		}
		change_variable = {
			name = kor_tutorial_var
			add = 30
		}
	}
}
