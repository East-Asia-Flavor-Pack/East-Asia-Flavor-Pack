				limit = { bureaucracy > 0 }

				}

					desc = "taiping_revolutionary_fervor_progress_bar_high_authority"

				limit = { government_legitimacy >= 70 }
				add = {
					desc = "taiping_revolutionary_fervor_progress_bar_high_legitimacy"

	}

	min_value = 0
	max_value = 100

taiping_internal_conflict_progress_bar = {
	name = "taiping_internal_conflict_progress_bar_name"
	desc = "taiping_internal_conflict_progress_bar_desc"

	default_bad = yes

	monthly_progress = {
		if = {
			limit = { has_journal_entry = je_taiping_internal_conflict }
			add = {
				desc = "taiping_internal_conflict_progress_bar_base"
				value = 0.1
			}
			if = {
				limit = { government_legitimacy < 50 }
				add = {
					desc = "taiping_internal_conflict_progress_bar_low_legitimacy"
					value = 0.3
				}
			}
			if = {
				limit = { government_legitimacy < 25 }
				add = {
					desc = "taiping_internal_conflict_progress_bar_very_low_legitimacy"
					value = 0.3
				}
			}
			if = {
				limit = { ruler.prominence < 25 }
				add = {
					desc = "taiping_internal_conflict_progress_bar_low_ruler_prominence"
					value = 0.3
				}
			}
			if = {
				limit = { ruler.prominence < 10 }
				add = {
					desc = "taiping_internal_conflict_progress_bar_very_low_ruler_prominence"
					value = 0.3
				}
			}
			add = {
				desc = "taiping_internal_conflict_progress_bar_ambitious_generals"
				value = {
					value = 0
					every_scope_character = {
						limit = {
							has_role_of_type = general
							is_ruler_of_own_country = no
							prominence > ROOT.ruler.prominence
						}
						add = 0.1
					}
				}
			}
			add = {
				desc = "taiping_internal_conflict_progress_bar_disloyal_ig_leaders"
				value = {
					value = 0
					every_interest_group = {
						limit = {
							is_in_government = yes
							ig_approval < 0
							leader = {
								is_ruler_of_own_country = no
								prominence > ROOT.ruler.prominence
							}
						}
						add = 0.1
					}
				}
			}
			if = {
				limit = {
					ig:ig_armed_forces ?= {
						ig_approval < 0
						leader = {
							is_ruler_of_own_country = no
							prominence > ROOT.ruler.prominence
						}
					}
				}
				add = {
					desc = "taiping_internal_conflict_progress_bar_disloyal_armed_forces"
					value = 0.2
				}
			}
			if = {
				limit = { authority < 0 }
				add = {
					desc = "taiping_internal_conflict_progress_bar_low_authority"
					value = 0.2
				}
			}
			if = {
				limit = { bureaucracy < 0 }
				add = {
					desc = "taiping_internal_conflict_progress_bar_low_bureaucracy"
					value = 0.2
				}
			}
			if = {
				limit = {
					je:je_taiping_house_on_sand ?= {
						"scripted_bar_progress(taiping_revolutionary_fervor_progress_bar)" < 40
					}
				}
				add = {
					desc = "taiping_internal_conflict_progress_bar_low_fervor"
					value = 0.2
				}
			}
			if = {
				limit = { authority > 100 }
				add = {
					desc = "taiping_internal_conflict_progress_bar_high_authority"
					value = -0.1
				}
			}
			if = {
				limit = { government_legitimacy > 70 }
				add = {
					desc = "taiping_internal_conflict_progress_bar_high_legitimacy"
					value = -0.2
				}
			}
			if = {
				limit = { ruler.prominence > 50 }
				add = {
					desc = "taiping_internal_conflict_progress_bar_high_ruler_prominence"
					value = -0.2
				}
			}
			if = {
				limit = { bureaucracy > 0 }
				add = {
					desc = "taiping_internal_conflict_progress_bar_positive_bureaucracy"
					value = -0.1
				}
			}
		}
	}

	start_value = 10
	min_value = 0
	max_value = 100
}
