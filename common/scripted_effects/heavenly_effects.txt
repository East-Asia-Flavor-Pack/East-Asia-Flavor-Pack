	#			state_region = scope:heavenly_kingdom_spawn.state_region
	#			count = 5
	#		}
	#	}
	#}
	if = {
		limit = {
			any_scope_state = {
				has_variable = cult_center
			}
		}
		every_scope_state = {
			limit = {
				has_variable = cult_center
			}
			save_scope_as = tpg_defector_state
			custom_tooltip = {
				text = taiping_state_defection
				set_state_owner = c:TPG
				set_state_type = incorporated
				add_culture_standard_of_living_modifier = {
					culture = cu:manchu
					days = long_modifier_time
					multiplier = -5
				}
			}
		}
		set_variable = { # 추가
			name = tpg_army_size_mul
			value = 0
		}
		while = {
			limit = {
				var:tpg_army_size_mul <= c:CHI.army_size
			}
			random_scope_state = { # 전체적으로 수정
				limit = {
					is_capital = no
					region = sr:region_south_china # 추가
					NOT = { 
						has_variable = cult_center
						state_region = {
							is_homeland = cu:manchu
						}
					}
					any_neighbouring_state = {
						has_variable = cult_center
					}
					# turmoil >= 0.1
					NOT = { state_region = s:STATE_YUNNAN } # 추가
					NOT = { state_region = s:STATE_GUIZHOU } # 추가
					NOT = { state_region = s:STATE_SICHUAN } # 추가
					NOT = { state_region = s:STATE_CHONGQING } # 추가
				}
				weight = { # 추가
					base = 1
					modifier = {
						turmoil > 0.1
						add = 1
					}
					modifier = {
						turmoil > 0.2
						add = 1
					}
					modifier = {
						turmoil > 0.3
						add = 1
					}
					modifier = {
						turmoil > 0.4
						add = 1
					}
					modifier = {
						turmoil > 0.5
						add = 1
					}
					modifier = {
						turmoil > 0.6
						add = 1
					}
					modifier = {
						turmoil > 0.7
						add = 1
					}
					modifier = {
						turmoil > 0.8
						add = 1
					}
					modifier = {
						turmoil > 0.9
						add = 1
					}
					modifier = {
						region = sr:region_south_china
						add = 1
					}
					modifier = {
						is_coastal = yes
						add = 1
					}
					modifier = {
						region = sr:region_south_china
						factor = 10
					}
					modifier = {
						is_coastal = yes
						factor = 2
					}
				}
				if = {
					limit = {
						this = {
							state_region = {
								OR = {
									is_homeland = cu:han
									is_homeland = cu:hakka
								}
							}
						}
					}
					save_scope_as = tpg_defector_state
					custom_tooltip = {
						text = taiping_state_defection
						set_state_owner = c:TPG
						set_state_type = incorporated
						set_variable = cult_center # 추가
						add_culture_standard_of_living_modifier = {
							culture = cu:manchu
							days = long_modifier_time
							multiplier = -5
						}
						c:TPG = {
							create_military_formation = {
								type = army
								hq_region = scope:tpg_defector_state.region
						
								combat_unit = {
									type = unit_type:combat_unit_type_irregular_infantry
									state_region = scope:tpg_defector_state.state_region
									count = 5
								}
							}
						}
					}
							
				}
				else = { # random_list 날림
					add_radicals_in_state = {
						value = very_large_radicals
						religion = rel:protestant
					}
					this.state_region = {
						add_claim = c:TPG
					}
				}
			}
			set_variable = { # 추가
				name = tpg_army_size_mul
				value = {
					value = c:TPG.army_size
					multiply = 5
				}
			}
		}
	}
	hidden_effect = {
		every_state = {
			limit = {
				owner = c:TPG
			}
			this.state_region = {
				add_claim = c:CHI
			}
		}
		every_state = { # 추가
			limit = {
				owner = c:CHI
			}
			this.state_region = {
				add_claim = c:TPG
			}
		}
	}
	hidden_effect = {
		every_state = {
			limit = {
				owner = c:TPG
			}
			add_modifier = {
				name = modifier_taiping_hiring_bonus
				months = 36 # 수정
				is_decaying = yes # 추가
			}
		}
	}
}
