eafp_kor_foreign_mission_cleanup = {
	if = {
		limit = { exists = var:eafp_kor_foreign_mission_leader_storage_var }
		var:eafp_kor_foreign_mission_leader_storage_var ?= {
			set_character_busy_and_immortal = no
			remove_variable = eafp_kor_foreign_mission_leader_var
		}
		remove_variable = eafp_kor_foreign_mission_leader_storage_var
	}
	if = {
		limit = { has_variable = eafp_kor_foreign_mission_abroad }
		remove_variable = eafp_kor_foreign_mission_abroad
	}
	remove_variable = eafp_kor_foreign_mission_world_tour
	remove_variable = eafp_kor_foreign_mission_extended_tour
	remove_variable = eafp_kor_foreign_mission_recalled
	remove_variable = eafp_kor_foreign_mission_en_route_home
	remove_variable = eafp_kor_foreign_mission_returned_home
	remove_variable = eafp_kor_foreign_mission_complete
}

eafp_kor_foreign_mission_pick_host_country = {
	if = {
		limit = {
			any_country = {
				NOT = { THIS = ROOT }
				NOT = { has_war_with = ROOT }
				OR = {
					country_rank >= rank_value:major_power
					has_technology_researched = railways
					has_technology_researched = electric_telegraph
					has_technology_researched = electrical_generation
					has_technology_researched = steel_frame_buildings
					has_technology_researched = central_banking
				}
			}
		}
		random_country = {
			limit = {
				NOT = { THIS = ROOT }
				NOT = { has_war_with = ROOT }
				OR = {
					country_rank >= rank_value:major_power
					has_technology_researched = railways
					has_technology_researched = electric_telegraph
					has_technology_researched = electrical_generation
					has_technology_researched = steel_frame_buildings
					has_technology_researched = central_banking
				}
			}
			save_scope_as = kor_foreign_mission_host_country
		}
	}
	else_if = {
		limit = {
			any_country = {
				NOT = { THIS = ROOT }
				NOT = { has_war_with = ROOT }
			}
		}
		random_country = {
			limit = {
				NOT = { THIS = ROOT }
				NOT = { has_war_with = ROOT }
			}
			save_scope_as = kor_foreign_mission_host_country
		}
	}
}
