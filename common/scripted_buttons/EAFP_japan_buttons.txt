japan_tutorial_button = {
	name = "japan_tutorial_button"
	desc = "japan_tutorial_button_desc"

	visible = {
		always = yes
	}

	possible = {
		root.var:japan_tutorial_var < 30
	}

	effect = {
		trigger_event = {
			id = eafp_japan.1
		}
		change_variable = {
			name = japan_tutorial_var
			add = 30
		}
	}
}

japan_sonnoha_decisions_button = {
	name = "japan_sonnoha_decisions_button"
	desc = "japan_sonnoha_decisions_button_desc"

	visible = {
		NOT = { has_variable = sonnoha_decisions_on }
	}

	possible = {
		NOT = { has_variable = sonnoha_decisions_on }
	}

	effect = {
		set_variable = {
			name = sonnoha_decisions_on
		}
	}

}

japan_hokkaido_button = {
	name = "japan_hokkaido_button"
	desc = "japan_hokkaido_button_desc"

	visible = {
		always = yes
	}

	possible = {
		s:STATE_HOKKAIDO.region_state:JAP = {
			not = { has_modifier = hokkaido_iju }
		}
	}

	effect = {
		s:STATE_HOKKAIDO.region_state:JAP = {
			add_modifier = {
				name = hokkaido_iju
				months = 12
			}
		}
		trigger_event = {
			id = eafp_japan.1001
		}
	}
}
