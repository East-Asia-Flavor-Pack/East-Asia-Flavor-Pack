boshin_war.6 = { # 의견일치
	type = country_event
	placement = root

	title = boshin_war.6.t
	desc = boshin_war.6.d
	flavor = boshin_war.6.f

	duration = 1

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	event_image = {
		video = "eafp_taisei_hokan"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	immediate = {
		if = { # 신정부 사라지는 버그 방지용
			limit = { NOT = { exists = c:NIP } }
			create_country = {
				tag = NIP
				origin = this
				province = p:x8EC4C9
				on_created = {
					change_relations = {
						country = c:JAP
						value = -50
					}
				}
			}
			c:NIP = {
				random_scope_state = {
					create_building={
						building="building_barracks"
						level=2
						reserves=1
					}
				}
				create_bidirectional_truce = {
					country = c:JAP
					months = 36
				}
				#ig:ig_landowners = {
				#	set_interest_group_name = ig_daimyo
				#}
				#ig:ig_devout = {
				#	set_interest_group_name = ig_kokugakusha
				#}
				#ig:ig_intelligentsia = {
				#	set_interest_group_name = ig_rangakusha
				#}
				#ig:ig_petty_bourgeoisie = {
				#	set_interest_group_name = ig_kashin
				#}
				#ig:ig_trade_unions = {
				#	set_interest_group_name = ig_shokunin
				#}
				#ig:ig_industrialists = {
				#	set_interest_group_name = ig_shonin
				#}
			}
		}
		every_country = {
			limit = {
				NOT = { THIS = c:NIP }
				has_variable = tobaku_han_var
			}
			c:NIP = {
				annex = PREV
			}
		}
		every_scope_state = {
			limit = {
				state_is_in_japan = yes
			}
			set_state_type = incorporated
		}
		c:NIP = {
			add_technology_researched = napoleonic_warfare
			add_technology_researched = general_staff
		}
		add_modifier = {
			name = modifier_japan_hiring_bonus
			months = 6
		}
		while = {
			count = 10
			create_character = {
				culture = cu:japanese
				is_general = yes
				hq = region_japan
				commander_rank = commander_rank_1
			}
		}
	}

	option = {
		name = boshin_war.6.a
		default_option = yes
	}
}
