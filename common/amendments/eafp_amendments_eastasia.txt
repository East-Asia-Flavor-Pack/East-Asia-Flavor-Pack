	allowed_laws = {
		law_landed_voting
		law_wealth_voting

		law_universal_suffrage
	}

	modifier = {
		country_legitimacy_govt_total_votes_add = -5
		country_legitimacy_govt_total_clout_add = 5

		country_legitimacy_ideological_incoherence_mult = 0.025
		country_electoral_confidence_impact_mult = -0.025
	}

	possible = {
		OR = {
			has_journal_entry = je_eastasian_constitution
			has_journal_entry = je_eastasian_constitution_active
		}
	}

	would_sponsor = {
		AND = {

		}
	}

		always = yes
	}

}
