korean_restoration_yangjeon_button_slow = {
	name = "yangjeon_button_slow"
	desc = "yangjeon_button_slow_desc"

	possible = {
		custom_tooltip = {
			text = not_already_yangjeon_button_slow
			NOT = { var:yangjeon_speed_var = 1 }
		}
		custom_tooltip = {
			text = recently_select_yangjeon_button
			NOT = { has_variable = yangjeon_button_timer_var }
		}
	}
	
	selected = {
		var:yangjeon_speed_var = 1
	}

	effect = {
		set_variable = {
			name = yangjeon_speed_var
			value = 1
		}
		set_variable = {
			name = yangjeon_button_timer
			days = 90
		}
		custom_tooltip = {
			text = yangjeon_button_slow_tt
		}
	}
}
