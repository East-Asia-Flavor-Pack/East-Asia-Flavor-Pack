first_opium_war.13 = {
	type = country_event
	placement = ROOT

	title = first_opium_war.13.t
	desc = first_opium_war.13.d
	flavor = first_opium_war.13.f

	duration = 3

	event_image = {
		video = "unspecific_politicians_arguing"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/asia/politician_parliament_motion"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	trigger = {
		THIS = c:GBR
		exists = c:CHI
		c:CHI = {
			has_variable = opium_wars_target
		}
		NOT = { is_diplomatic_play_enemy_of = c:CHI }
	}

	immediate = {
		save_scope_as = opium_wars_aggressor_country
		c:CHI = { save_scope_as = opium_wars_target_country }
		cu:han = { save_scope_as = opium_addicted_culture }
	}

	option = {
		name = first_opium_war.13.a
		default_option = yes
		set_variable = opium_wars_aggressor
		if = {
			limit = { has_truce_with = c:CHI }
			end_truce = c:CHI
		}
		change_relations = {
			country = c:CHI
			value = -30
		}
		add_journal_entry = {
			type = je_opium_wars
			target = c:CHI
		}
		add_involvement = {
			strategic_region = sr:region_south_china
			value = 2500
		}
		create_diplomatic_play = {
			name = first_opium_war
			target_country = c:CHI
			type = dp_first_opium_war
			add_war_goal = {
				holder = c:GBR
				type = open_market
				target_country = c:CHI
				primary_demand = yes
			}
		}
		random_diplomatic_play = {
			limit = {
				initiator = c:GBR
				target = c:CHI
				is_diplomatic_play_type = dp_first_opium_war
			}
			add_escalation = 40
			add_diplomatic_play_war_support = { target = c:GBR value = 100 }
			add_diplomatic_play_war_support = { target = c:CHI value = 100 }
		}
		c:CHI = {
			set_secret_goal = {
				country = root
				secret_goal = antagonize
			}
		}
		set_secret_goal = {
			country = c:CHI
			secret_goal = conquer
		}
		ai_chance = {
			base = 1
		}
	}
}
