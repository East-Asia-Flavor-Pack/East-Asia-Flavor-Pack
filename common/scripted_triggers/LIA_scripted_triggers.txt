##다양한 트리거들.

#수교 여부 확인. 상대국( LIA_NATION_OTHER ) 과의 수교관계 존재 여부를 반환함.
lia_trigger_nation_is_signed_with = {
	# Parameters
	$LIA_NATION_OTHER$ = {
		save_temporary_scope_as = lia_nation_other
	}

	# 대조
	trigger_if = {
		limit = {
			exists = this
			exists = scope:lia_nation_other
			##핵심 조건문
		}
		custom_tooltip = {
			text = lia_eafp_treaty_signed_yes_tt
			#always = yes
		}
	}
	trigger_else = {
		custom_tooltip = {
			text = lia_eafp_treaty_signed_no_tt
			#always = yes
		}
	}
}
