eafp_kor_manchuria_check_journals = {
	if = {
		limit = {
			c:KOR ?= this
			eafp_kor_manchuria_policy_prerequisite = yes
			NOT = { is_involved_in_journal_entry = je_eafp_manchurian_policy }
			NOT = { has_inactive_journal_entry = je_eafp_manchurian_policy }
			NOT = { has_variable = eafp_manchurian_claims_granted_var }
			NOT = { has_variable = eafp_manchurian_policy_complete_var }
		}
		add_contextless_journal_entry = je_eafp_manchurian_policy
	}
	if = {
		limit = {
			c:KOR ?= this
			eafp_manchuria_owns_all_manchurian_states = yes
			has_variable = eafp_manchurian_claims_granted_var
			NOT = { is_involved_in_journal_entry = je_eafp_manchurian_management }
			NOT = { has_inactive_journal_entry = je_eafp_manchurian_management }
			NOT = { has_variable = eafp_manchurian_policy_complete_var }
		}
		add_contextless_journal_entry = je_eafp_manchurian_management
	}
}

eafp_manchurian_policy_add_involved_countries = {
	scope:journal_entry = {
		every_scope_je_involved = {
			save_temporary_scope_as = eafp_manchurian_policy_actor_scope
			if = {
				limit = {
					NOT = {
						OR = {
							c:KOR ?= this
							c:CHI ?= this
							any_scope_state = {
								eafp_manchuria_is_core_manchurian_state = yes
								owner = scope:eafp_manchurian_policy_actor_scope
							}
						}
					}
				}
				scope:journal_entry = { remove_involved_country = prev }
			}
		}
	}
	every_country = {
		save_temporary_scope_as = eafp_manchurian_policy_actor_scope
		if = {
			limit = {
				NOT = { is_involved_in_journal_entry = je_eafp_manchurian_policy }
				OR = {
					c:KOR ?= this
					c:CHI ?= this
					any_scope_state = {
						eafp_manchuria_is_core_manchurian_state = yes
						owner = scope:eafp_manchurian_policy_actor_scope
					}
				}
			}
			scope:journal_entry = { add_involved_country = prev }
		}
	}
}
