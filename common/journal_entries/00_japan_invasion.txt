je_japan_invasion = {#정일론
     icon = "gfx/interface/icons/event_icons/event_scales.dds"

     complete = {
		ROOT = {
			army_size >= 180
			navy_size >= 40
              }

        c:JAP = {
		country_rank < rank_value:great_power
		}
	}

	on_complete = {
		 trigger_event = {
			id = japan_invasion.1
		 }
	}
   
}

je_japan_invasion_second = {#측량
    icon = "gfx/interface/icons/event_icons/event_scales.dds"

immediate = {
		set_variable = {
			name = japan_invasion_second_var
			value = 0
		}
	}


complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
        custom_tooltip = {
            text = shizuoka_treaty_tt
        has_global_variable = shizuoka_treaty
        }
		
	}

	on_complete = {
		trigger_event = {
			id = japan_invasion.7
		}
	}



    on_monthly_pulse = {
        on_monthly_pulse = {
		random_events = {
			100 = 0
			5 = japan_invasion.2
		}
	}
		effect = {
			change_variable = {
					name = japan_invasion_second_var
					add = 1
				}
		}
	}

	current_value = {
		value = root.var:japan_invasion_second_var
	}

	goal_add_value = {
	      value = 30
	}

	progressbar = yes

}
