		video = "europenorthamerica_opium_smoker"

	on_opened_soundeffect = "event:/SFX/Events/europenorthamerica/opium_smoker"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	option = {
		name = first_opium_war.1.a
		default_option = yes
		set_variable = { name = eafp_opium_policy_score value = -2 }
		change_variable = { name = eafp_opium_crisis_pressure add = 20 }
		add_banned_goods = g:opium
		trigger_event = { id = first_opium_war.10 days = { 30 60 } }
		ai_chance = { base = 1 }
	}

	option = {
		name = first_opium_war.1.b
		trigger = { is_ai = no }
		set_variable = { name = eafp_opium_policy_score value = 2 }

			remove_banned_goods = g:opium

		trigger_event = { id = first_opium_war.20 days = { 30 60 } }
		ai_chance = { base = 0 }
	}

	option = {
		name = first_opium_war.1.c
		trigger = { is_ai = no }
		set_variable = { name = eafp_opium_policy_score value = 0 }
		trigger_event = { id = first_opium_war.31 days = { 45 90 } }
		ai_chance = { base = 0 }
	}
}
