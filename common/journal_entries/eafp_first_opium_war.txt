je_first_opium_war = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"
	group = je_group_foreign_affairs
	complete = {
		OR = {
			scope:target ?= {
				has_law_or_variant = law_type:law_free_trade
			}
			has_treaty_foreign_investment_rights_with = { TARGET = scope:target }
			any_scope_state = {
				is_treaty_port = yes
				state_region = {
					any_scope_state = {
						owner = scope:target
					}
				}
			}
		}
	}
	on_complete = {
		c:CHI = {
			trigger_event = {
				id = first_opium_war.153
				days = 5
			}
		}
		c:GBR = {
			trigger_event = {
				id = first_opium_war.154
				days = 5
			}
		}
	}
	fail = {
		NOT = {
			OR = {
				scope:target ?= {
					has_law_or_variant = law_type:law_free_trade
				}
				has_treaty_foreign_investment_rights_with = { TARGET = scope:target }
				any_scope_state = {
					is_treaty_port = yes
					state_region = {
						any_scope_state = {
							owner = scope:target
						}
					}
				}
			}
		}
		NOT = {
			any_diplomatic_play = {
				initiator = c:GBR
				target = c:CHI
				is_diplomatic_play_type = dp_first_opium_war
			}
		}
	}
	invalid = {
		NOT = { exists = scope:target }
	}
	on_invalid = {
		remove_variable = opium_wars_aggressor
	}
	on_fail = {
		c:CHI = {
			trigger_event = {
				id = first_opium_war.155
				days = 5
			}
		}
		c:GBR = {
			trigger_event = {
				id = first_opium_war.156
				days = 5
			}
		}
	}
	weight = 10000
	should_be_pinned_by_default_uninvolved_or_context = yes
}

je_first_opium_war_legalization = {
	icon = "gfx/interface/icons/goods_icons/opium.dds"
	group = je_group_qing

	complete = {
		market = { mg:opium = { market_goods_pricier <= 0.2 } }
		scaled_gold_reserves >= 0.2
		any_interest_group = {
			count = all
			ig_approval >= 0
		}
	}
	on_complete = {
		trigger_event = { id = first_opium_war.251 popup = yes }
	}
	invalid = {
		OR = {
			NOT = { exists = c:CHI }
			NOT = { THIS = c:CHI }
			has_variable = opium_wars_target
			has_variable = opium_wars_gave_in
			AND = {
				OR = {
					has_law = law_type:law_canton_system
					has_law = law_type:law_isolationism
				}
				active_law:lawgroup_trade_policy ?= {
					NOT = {
						has_amendment = amendment_type:amendment_first_opium_war_legalization
					}
				}
			}
		}
	}
	on_invalid = {
		if = {
			limit = {
				NOT = { has_variable = opium_wars_target }
				NOT = { has_variable = opium_wars_gave_in }
				OR = {
					has_law = law_type:law_canton_system
					has_law = law_type:law_isolationism
				}
				active_law:lawgroup_trade_policy ?= {
					NOT = {
						has_amendment = amendment_type:amendment_first_opium_war_legalization
					}
				}
			}
			add_modifier = {
				name = eafp_first_opium_war_legalization_repealed_decay
				days = normal_modifier_time
				is_decaying = yes
			}
			c:GBR = { trigger_event = { id = first_opium_war.112 popup = yes } }
		}
	}
	on_timeout = {
		trigger_event = { id = first_opium_war.201 popup = yes }
	}
	timeout = 3600
	weight = 1000
	should_be_pinned_by_default_uninvolved_or_context = yes
}
