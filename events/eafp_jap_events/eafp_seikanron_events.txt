seikanron_events.6 = { # 장교들의 사직
	type = country_event
	placement = ROOT

	title = seikanron_events.6.t
	desc = seikanron_events.6.d
	flavor = seikanron_events.6.f

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	event_image = { 
		video = "asia_confucianism_shinto"
	}
	
	duration = 3
	
	immediate = {
		random_scope_character = {
			limit = {
				has_role = general
				is_ruler = no
				is_heir = no
			}
			save_scope_as = character_scope
		}
	}

	trigger = {
		any_political_lobby = {
			is_political_lobby_type = lobby_anti_country
			target = c:KOR
			appeasement <= 0
		}
		any_scope_character = {
			has_role = general
			is_ruler = no
			is_heir = no
		}
	}
	
	option = {
		name = seikanron_events.6.a

		default_option = yes

		if = {
			limit = {
				NOT = {
					ig:ig_armed_forces = {
						any_political_lobby = {
							is_political_lobby_type = lobby_anti_country
							target = c:KOR
						}
					}
				}
			}
			random_political_lobby = {
				limit = {
					is_political_lobby_type = lobby_anti_country
					target = c:KOR
				}
				add_lobby_member = ROOT.ig:ig_armed_forces
			}
		}
		every_military_formation = {
			add_organization = {
				value = organization
				multiply = -0.25
			}
		}
		ig:ig_armed_forces = {
			add_modifier = {
				name = seikan_officers_resign_modifier
				months = normal_modifier_time
			}
		}

		ai_chance = {
			base = 1
		}
	}

	option = {

						any_political_lobby = {
							is_political_lobby_type = lobby_anti_country
							target = c:KOR
						}
					}
				}
			}
			random_political_lobby = {
				limit = {
					is_political_lobby_type = lobby_anti_country
					target = c:KOR
				}
				add_lobby_member = ROOT.ig:ig_armed_forces
			}
		}
		random_political_lobby = {
			limit = {
				is_political_lobby_type = lobby_anti_country
				target = c:KOR
			}
			change_appeasement = {
				amount = -1
				factor = appeasement_special_events_negative
			}
		}

		ai_chance = {
			base = 1

seikanron_events.8 = { # 정한파 자객
	type = country_event
	placement = ROOT

	title = seikanron_events.8.t
	desc = seikanron_events.8.d
	flavor = seikanron_events.8.f

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	event_image = { 
		video = "asia_confucianism_shinto"
	}
	
	duration = 3
	
	immediate = {
		if = {
			limit = {
				any_political_lobby = {
					is_political_lobby_type = lobby_pro_country
					target = c:KOR
				}
			}
			random_scope_character = {
				limit = {
					interest_group = {
						any_political_lobby = {
							is_political_lobby_type = lobby_pro_country
							target = c:KOR
						}
					}
				}
				save_scope_as = character_scope
			}
		}
		else = {
			random_scope_character = {
				limit = {
					NOT = {
						interest_group = {
							any_political_lobby = {
								is_political_lobby_type = lobby_anti_country
								target = c:KOR
							}
						}
					}
				}
				save_scope_as = character_scope
			}
		}
	}

	trigger = {
		OR = {
			AND = {
				any_political_lobby = {
					is_political_lobby_type = lobby_anti_country
					target = c:KOR
					appeasement <= 0
				}
				any_political_lobby = {
					is_political_lobby_type = lobby_pro_country
					target = c:KOR
				}
			}
			AND = {
				any_political_lobby = {
					is_political_lobby_type = lobby_anti_country
					target = c:KOR
					appeasement <= -3
				}
				any_interest_group = {
					NOT = {
						any_political_lobby = {
							is_political_lobby_type = lobby_anti_country
							target = c:KOR
						}
					}
				}
			}
		}
	}
	
	option = {
		name = seikanron_events.8.a

		default_option = yes

		scope:character_scope = {
			add_modifier = {
				name = seikan_assassins_attack_character_modifier
				months = normal_modifier_time
			}
		}
		scope:character_scope.interest_group = {
			add_modifier = {
				name = seikan_assassins_attack_modifier
				months = normal_modifier_time
			}
		}
		random_political_lobby = {
			limit = {
				is_political_lobby_type = lobby_anti_country
				target = c:KOR
			}
			change_appeasement = {
				amount = -1
				factor = appeasement_special_events_negative
			}
		}

		ai_chance = {
			base = 1
		}
	}
}

	placement = scope:state_scope

	title = seikanron_events.14.t
	desc = seikanron_events.14.d
	flavor = seikanron_events.14.f

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	event_image = { 
		video = "asia_confucianism_shinto"
	}
	
	duration = 3
	
	immediate = {
		random_scope_state = {
			limit = { is_incorporated = yes }
			save_scope_as = state_scope
		}
		c:KOR = {
			save_scope_as = korea_scope
		}
	}

	trigger = {
		any_political_lobby = {
			is_political_lobby_type = lobby_anti_country
			target = c:KOR
			appeasement <= -4
		}
	}
	
	option = {
		name = seikanron_events.14.a

		default_option = yes

		random_interest_group = {
			limit = {
				any_political_lobby = {
					is_political_lobby_type = lobby_anti_country
					target = c:KOR
				}
			}
			add_modifier = {
				name = captured_and_executed_modifier
				months = normal_modifier_time
			}
		}

		ai_chance = {
			base = 1
		}
	}

	option = {
		name = seikanron_events.14.b

		random_political_lobby = {
			limit = {
				is_political_lobby_type = lobby_anti_country
				target = c:KOR
			}
			change_appeasement = {
				amount = -1
				factor = appeasement_special_events_positive
			}
		}

		ai_chance = {
			base = 1
		}
	}
}

seikanron_events.15 = { # 확전 요구
	type = country_event
	placement = scope:state_scope

	title = seikanron_events.15.t
	desc = seikanron_events.15.d
	flavor = seikanron_events.15.f

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	event_image = { 
		video = "asia_confucianism_shinto"
	}
	
	duration = 3
	
	immediate = {
		c:KOR = { save_scope_as = korea_scope }
	}

	trigger = {
		is_diplomatic_play_enemy_of = c:KOR
		NOT = {
			play_participant_has_war_goal_of_type_against = {
				type = annex_country
				target = c:KOR
			}
		}
	}
	
	option = {
		name = seikanron_events.15.a

		default_option = yes

		random_political_lobby = {
			limit = {
				is_political_lobby_type = lobby_anti_country
				target = c:KOR
			}
			change_appeasement = {
				amount = -5
				factor = appeasement_special_events_positive
			}
		}

		ai_chance = {
			base = 1
		}
	}

	option = {
		name = seikanron_events.15.b

		random_diplomatic_play = {
			limit = {
				OR = {
					AND = {
						OR = {
							target_is = ROOT
							any_scope_target_ally = { THIS = ROOT }
						}
						OR = {
							initiator_is = c:KOR
							any_scope_initiator_ally = { THIS = c:KOR }
						}
					}
					AND = {
						OR = {
							target_is = c:KOR
							any_scope_target_ally = { THIS = c:KOR }
						}
						OR = {
							initiator_is = ROOT
							any_scope_initiator_ally = { THIS = ROOT }
						}
					}
				}
			}
			add_war_goal = {
				holder = ROOT
				target_country = c:KOR
				type = annex_country
				primary_demand = yes
			}
		}

		ai_chance = {
			base = 1
		}
	}
}
