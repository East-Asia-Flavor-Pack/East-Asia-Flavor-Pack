namespace = chi_war_dynamism

chi_war_dynamism.1 = { # 아직 변동치 있고 조건 만족하면 주 소유 넘김
	type = state_event
	hidden = yes

	trigger = {
		has_modifier = chi_war_dynamism_modifier
		controller = {
			OR = {
				THIS = c:CHI
				country_has_primary_culture = cu:han
			}
			has_war_with = ROOT.owner
			any_neighbouring_state = { owner = ROOT.owner }
		}
	}

	immediate = {
		owner = {
			random_scope_war = {
				limit = {
					is_war_participant = root.controller
				}
				add_war_war_support = {
					target = ROOT.controller
					value = 10
				}
				add_war_war_support = {
					target = ROOT.owner
					value = -5
				}
			}
		}
		remove_modifier = chi_war_dynamism_modifier
		set_state_owner = controller
		set_state_type = incorporated
	}
}
