			name = donghak_civil_war

			type = dp_annex_war

				holder = c:ABC
				type = annex_country
				target_country = c:KOR
			}
		}
	}

		name = donghak_je_events.1.a
		default_option = yes

		set_variable = union_side_donghak
		hidden_effect = { 
			scope:abc_scope = {
				ig:ig_devout = {
			leader = { set_ideology = ideology:ideology_donghak }
		}
		ig:ig_rural_folk = {
			leader = { set_ideology = ideology:ideology_donghak }
		}
			 create_character = {
		 # 전봉준
                first_name = "Bongjun"
                last_name = "Jeon"
                ruler = yes
                birth_date = 1800.1.10
                interest_group = ig_rural_folk 
                ideology = ideology_donghak
                      traits = {
                    romantic
		   }
		  }
	}
	}
		if = {
			limit = { exists = c:ABC }
			add_journal_entry = {
				type = je_donghak_war
				target = scope:abc_scope

			}
		}
	}
	option = {
		name = donghak_je_events.1.b

		trigger = { is_player = yes }

		
		set_variable = union_side_donghak
		set_variable = donghak_gun
		play_as = scope:abc_scope
		scope:abc_scope = {
		if = {
			limit = { exists = c:ABC }
			add_journal_entry = {
				type = je_donghak_war_2
				target = scope:abc_scope
			}
			}
		hidden_effect = { 
			ig:ig_devout = {
			leader = { set_ideology = ideology:ideology_donghak }
		}
		ig:ig_rural_folk = {
			leader = { set_ideology = ideology:ideology_donghak }

donghak_je_events.4 = {
	type = country_event
	event_image = {
		texture = "gfx/event_pictures/champi3.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	#on_opened_soundeffect = "event:/SFX/Events/europenorthamerica/capitalists_meeting"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	title = donghak_je_events.4.t
	desc = donghak_je_events.4.d
	flavor = donghak_je_events.4.f

	duration = 3

	trigger = {
		# 동학농민운동으로 일어남
		exists = c:ABC
		exists = c:KOR
		NOT = { has_law = law_type:law_tenant_farmers }
		any_diplomatic_play = {
			AND = {
				any_scope_play_involved = { this = c:ABC }
				any_scope_play_involved = { this = c:KOR }
			}
		}
		NOT = { 
			any_interest_group = {
				has_modifier = delayed_serfdom_banned
			}
		}
		has_variable = union_side_donghak
	}

	immediate = {
		law_type:law_tenant_farmers = {
			save_scope_as = abolition_law
		}
	}

	option = {
		name = donghak_je_events.4.a 
		default_option = yes
		activate_law = law_type:law_tenant_farmers

		every_interest_group = {
			limit = {
				law_stance = {
					law = law_type:law_tenant_farmers
					value > neutral
				}
			}
			add_modifier = {
				name = enacted_serfdom_banned # +1 opinion
				months = normal_modifier_time
			}
		}
	}

	option = {
		name = donghak_je_events.4.b

		every_interest_group = {
			limit = {
				law_stance = {
					law = law_type:law_tenant_farmers
					value > neutral
				}
			}
			add_modifier = {
				name = delayed_serfdom_banned
				months = normal_modifier_time
			}
		}
	}
}
