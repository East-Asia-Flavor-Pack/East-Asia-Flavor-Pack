			if = {
				limit = {
					NOT = {
						has_amendment = amendment_type:amendment_first_opium_war_legalization

			add_diplomatic_play_war_support = { target = c:CHI value = 100 }
		}
		c:CHI = {
			set_variable = opium_wars_target

				country = root
				secret_goal = antagonize
			}
		}

			country = c:CHI
			secret_goal = conquer
		}

	}
