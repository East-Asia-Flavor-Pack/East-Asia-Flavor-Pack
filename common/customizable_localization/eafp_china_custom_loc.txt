china_manchu_power_custom_loc = {
	type = country
	text = {
		trigger = {
			var:china_manchu_power_var >= 80
		}
		localization_key = "$tariff_level_very_high$"
	}
	text = {
		trigger = {
			var:china_manchu_power_var >= 60
		}
		localization_key = "$tariff_level_high$"
	}
	text = {
		trigger = {
			var:china_manchu_power_var >= 40
		}
		localization_key = "$tariff_level_medium$"
	}
	text = {
		trigger = {
			var:china_manchu_power_var >= 20
		}
		localization_key = "$tariff_level_low$"
	}
	text = {
		localization_key = "$tariff_level_very_low$"
	}
}
