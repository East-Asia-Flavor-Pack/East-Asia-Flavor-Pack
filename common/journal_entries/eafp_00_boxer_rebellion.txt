je_boxer_rebellion_pre = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_qing

	is_shown_when_inactive = {
		NOT = { has_global_variable = boxer_rebellion_happened }
		has_journal_entry = je_warlord_china
	}

	possible = {
		has_journal_entry = je_warlord_china
		OR = {
			has_technology_researched = political_agitation
			AND = {
				has_technology_researched = nationalism
				custom_tooltip = {
					text = china_explosion_point_25_tt2
					has_variable = china_warlord_explosion_sum
					var:china_warlord_explosion_sum <= 0.25
				}
			}
		}
		any_state = {
			state_region = { is_homeland = cu:han }
			NOT = { owner = PREV }
		}
	}

	immediate = {
		set_global_variable = boxer_rebellion_happened
		trigger_event = { id = boxer_rebellion_events.1 } 
	}

	complete = {
		OR = {
			has_journal_entry = je_boxer_rebellion_western
			has_journal_entry = je_boxer_rebellion_bxr
		}
	}

	on_monthly_pulse = {
		random_events = {
			10 = 0
			2 = boxer_rebellion_events.101
			1 = boxer_rebellion_events.102
			3 = boxer_rebellion_events.103
		}
	}
	weight = 1000
	should_be_pinned_by_default = yes
}

je_boxer_rebellion_western = { # 청나라가 서양 열강 편을 듦.
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_global_international_situations

	should_be_involved = {
		OR = {
			c:CHI ?= THIS
			c:BXR ?= THIS
			AND = {
				country_rank >= rank_value:great_power
				OR = {
					has_interest_marker_in_region = sr:region_north_china
					has_interest_marker_in_region = sr:region_south_china
				}
			}
		}
	}

	should_show_when_not_involved = {
		any_country = {
			is_involved_in_journal_entry = je_boxer_rebellion
		}
		country_rank >= rank_value:great_power
		OR = {
			has_interest_marker_in_region = sr:region_north_china
			has_interest_marker_in_region = sr:region_south_china
		}
	}

	immediate = {
		c:BXR = {
			save_scope_as = taiping_scope # 텍스트 돌려쓰려고.
		}
	}

	complete = {
		custom_tooltip = {
			text = taiping_destroyed
			hidden_trigger = {
				NOT = {
					exists = c:BXR
				}
			}
		}
	}

	on_complete = {
		c:CHI ?= {
			trigger_event = { id = boxer_rebellion_events.106 }
		}
	}
	
	fail = {
		exists = c:BXR
		OR = {
			c:CHI ?= { has_truce_with = c:BXR }
			NOT = { exists = c:CHI }
		}
	}
	on_fail = {
		c:CHI ?= {
			trigger_event = { id = boxer_rebellion_events.3 } # The Qing and the Han
		}
	}

	weight = 1000
	should_be_pinned_by_default = yes
}
