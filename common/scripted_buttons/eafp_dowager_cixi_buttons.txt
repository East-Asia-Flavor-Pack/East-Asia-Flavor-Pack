remove_cixi_button = {
	name = "remove_cixi_button"
	desc = "remove_cixi_button_desc"

	visible = {
		always = yes
	}

	possible = {
		NOT = { has_variable = remove_cixi_temp }
	}

	effect = {
		custom_tooltip = {
			text = remove_cixi_tt
			set_variable = {
				name = remove_cixi_temp
				value = {
					value = ruler.age
					multiply = 2
					add = 10
					add = {
						value = var:cixi_coup_d
						multiply = -100
					}
				}
			}
			random_list = { # 테스트 todo
				100 = {
					modifier = {
						add = {
							value = var:remove_cixi_temp
							multiply = -1
						}
					}
					trigger_event = {
						id = dowager_cixi_events.7
						days = { 30 60 }
					}
				}
				0 = {
					modifier = {
						add = var:remove_cixi_temp
					}
					trigger_event = {
						id = dowager_cixi_events.4
						days = { 30 60 }
					}
				}
			}
			set_variable = {
				name = remove_cixi_temp
				months = 12
			}
		}
	}
}
