#왕이나 왕비가 죽으면 "인산" 변수가 활성화 되면서 고두례를 1회 진행할 수 있음. 왕권 강화용 디시전

judarye_decision = { 
	is_shown = {
		exists = c:KOR
		c:KOR = THIS
		has_variable = insan_var
	}

	possible = {
		exists = c:KOR
		c:KOR = THIS
		has_variable = insan_var
	}

	when_taken = {
		remove_variable = insan_var
        trigger_event = {
            id = korea_main.10
        }
	}
	
	ai_chance = {
		base = 0
    }	

}

kwongang_decision = { #권강

	is_shown = {
		exists = c:KOR
		c:KOR = THIS
		NOT = { has_variable = kwongang_decision_var }
		ROOT = {
			ruler = {
				has_trait = trait_child
			}
		}
	}

	possible = {
		exists = c:KOR
		c:KOR = THIS
		NOT = { has_variable = kwongang_decision_var }
		ROOT = {
			ruler = {
				has_trait = trait_child
			}
		}
	}

	when_taken = {
		set_variable = {
			name = kwongang_decision_var
			months = 36
		}
        trigger_event = {
            id = korea_main.11
        }
	}
	
	ai_chance = {
		base = 0
    }	

}
