boshin_war_peace_sabaku_legitimacy_progress_bar = {
 	name = "boshin_war_peace_sabaku_legitimacy_progress_bar_name"
 	desc = "boshin_war_peace_sabaku_legitimacy_progress_bar_desc"

 	double_sided_bad = yes

 	weekly_progress = {
		add = {
			desc = "boshin_war_peace_legitimacy_progress_bar_from_JAP_loyalists"
			value = scope:JAP_scope.loyalist_fraction
		}
		subtract = {
			desc = "boshin_war_peace_legitimacy_progress_bar_from_JAP_radicals"
			value = scope:JAP_scope.radical_fraction
		}

		subtract = {
			desc = "boshin_war_peace_legitimacy_progress_bar_from_NIP_loyalists"
			value = scope:NIP_scope.loyalist_fraction
		}
		add = {
			desc = "boshin_war_peace_legitimacy_progress_bar_from_NIP_radicals"
			value = scope:NIP_scope.radical_fraction
		}
	}

	monthly_progress = {
		add = {
			desc = "boshin_war_peace_legitimacy_progress_bar_from_JAP_offer_embassy"
			value = {
				scope:JAP_scope = {
					every_scope_treaty = {
						limit = {
							any_scope_article = {
								has_type = offer_embassy
								target_country = scope:JAP_scope
							}
						}
						add = 1
					}
				}
			}
		}
		subtract = {
			desc = "boshin_war_peace_legitimacy_progress_bar_from_NIP_offer_embassy"
			value = {
				scope:NIP_scope = {
					every_scope_treaty = {
						limit = {
							any_scope_article = {
								has_type = offer_embassy
								target_country = scope:NIP_scope
							}
						}
						add = 1
					}
				}
			}
		}
	}

 	start_value = 50
 	min_value = 0
 	max_value = 100
}
