gg_korea.4 = { # Donghak Petition
	type = country_event
	placement = ROOT

	title = {
		first_valid = {
			triggered_desc = {
				desc = gg_korea.4.ta
				trigger = {
					scope:petition_state = {
						state_region = s:STATE_YANGHO
					}
				}
			}
			triggered_desc = {
				desc = gg_korea.4.tb
				trigger = {
					scope:petition_state = {
						NOT = {
							state_region = s:STATE_YANGHO
						}
					}
				}
			}
		}
	}
	desc = gg_korea.4.d
	flavor = gg_korea.4.f

	dlc = dlc010

	event_image = {
		video = "ep1_donghak_rebellion"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	duration = 3

	trigger = {
		NOT = {
			has_variable = donghak_petition_event_var
		}
		radical_fraction = {
			value >= 0.2
		}
		ig:ig_rural_folk = {
			is_marginal = no
			is_in_government = no
		}
	}

	immediate = {
		ig:ig_rural_folk = {
			save_scope_as = donghak_ig
		}
		if = {
			limit = {
				any_scope_state = {
					state_region = s:STATE_YANGHO
					radical_fraction = {
						value >= 0.1
					}
				}
			}
			random_scope_state = {
				limit = {
					state_region = s:STATE_YANGHO
					radical_fraction = {
						value >= 0.1
					}
				}
				save_scope_as = petition_state
			}
		}
		else = {
			random_scope_state = {
				limit = {
					radical_fraction = {
						value >= 0.1
					}
				}
				save_scope_as = petition_state
			}
		}
		set_variable = donghak_petition_event_var
	}

	option = {
		name = gg_korea.4.a
		default_option = yes
		scope:donghak_ig = {
			add_modifier = {
				name = modifier_donghak_petition_rejected
				days = normal_modifier_time
			}
		}
		scope:petition_state = {
			add_radicals_in_state = {
				value = medium_radicals
				pop_type = peasants
			}
		}
	}

	option = {
		name = gg_korea.4.b
		add_journal_entry = { type = je_gyojo_shinwon }
	}
}
