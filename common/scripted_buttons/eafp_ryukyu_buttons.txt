	name = button_je_eafp_ryukyu_dispatch_envoys
	desc = button_je_eafp_ryukyu_dispatch_envoys_desc

	visible = {
		c:KOR ?= this
		has_journal_entry = je_ryukyu_rivalry
		has_variable = eafp_ryukyu_intervention_var
	}

	cooldown = { days = 365 }

	possible = {
		exists = c:RYU
		custom_tooltip = {
			text = eafp_ryukyu_independence_progress_not_complete_tt

				"scripted_bar_progress(je_ryukyu_rivalry_progress_bar)" < 100
			}

		je:je_ryukyu_rivalry = {
			add_progress = {
				value = 10

			}
		}

			change_relations = {
				country = ROOT
				value = -10
			}
		}
		c:CHI ?= {
			change_relations = {
				country = ROOT
				value = -5
			}
		}
		custom_tooltip = eafp_ryukyu_dispatch_envoys_effect_tt
	}

	ai_chance = {
		value = 5
	}
}
