chi_war_dynamism_BEIJING_button = {
	name = "chi_war_dynamism_BEIJING_button"
	desc = "chi_war_dynamism_BEIJING_button_desc"

	visible = {
		any_scope_state = {
			state_region = s:STATE_BEIJING
			controller = {
				NOT = { THIS = ROOT }
				OR = {
					THIS = c:CHI
					country_has_primary_culture = cu:han
				}
			}
		}
	}

	possible = {
		any_scope_state = {
			state_region = s:STATE_BEIJING
			controller = {
				OR = {
					THIS = c:CHI
					country_has_primary_culture = cu:han
				}
				has_war_with = ROOT
				any_neighbouring_state = { owner = ROOT }
			}
			NOT = { has_modifier = chi_war_dynamism_modifier }
		}
	}

	effect = {
		custom_tooltip = {
			text = chi_war_dynamism_BEIJING_tt
			random_scope_state = {
				limit = {
					state_region = s:STATE_BEIJING
					controller = {
						OR = {
							THIS = c:CHI
							country_has_primary_culture = cu:han
						}
						has_war_with = ROOT
						any_neighbouring_state = { owner = ROOT }
					}
				}
				add_modifier = chi_war_dynamism_modifier
				trigger_event = { id = chi_war_dynamism.1 days = 30 }
			}
		}
	}
}
