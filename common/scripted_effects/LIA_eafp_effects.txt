lia_effect_treaty_relation_set_simp = {
	# Parameters
	$LIA_NATION_1$ = {
		save_scope_as = lia_treaty_nation_1
	}
	$LIA_NATION_2$ = {
		save_scope_as = lia_treaty_nation_2
	}

	# 수교국을 대상으로 다시 발동될경우, 기존 조약 관계 말소.
	scope:lia_treaty_nation_1 = {
		if = {
			limit = {
				lia_trigger_nation_is_signed_with = {
					LIA_NATION_OTHER = scope:lia_treaty_nation_2
				}
			}
			set_variable = {
				name = lia_var_equality_num
				value = -3
			}
			while = {
				limit = {
					var:lia_var_equality_num <= 3
				}
				lia_effect_treaty_relation_clear = {
					LIA_EQUALITY_NUM = var:lia_var_equality_num
					LIA_NATION_CLEAR = scope:lia_treaty_nation_2
				}
				change_variable = {
					name = lia_var_equality_num
					add = 1
				}
			}
			remove_variable = lia_var_equality_num
		}
	}
	scope:lia_treaty_nation_2 = {
		if = {
			limit = {
				lia_trigger_nation_is_signed_with = {
					LIA_NATION_OTHER = scope:lia_treaty_nation_1
				}
			}
			set_variable = {
				name = lia_var_equality_num
				value = -3
			}
			while = {
				limit = {
					var:lia_var_equality_num <= 3
				}
				lia_effect_treaty_relation_clear = {
					LIA_EQUALITY_NUM = var:lia_var_equality_num
					LIA_NATION_CLEAR = scope:lia_treaty_nation_1
				}
				change_variable = {
					name = lia_var_equality_num
					add = 1
				}
			}
			remove_variable = lia_var_equality_num
		}
	}

	# 조약 체결/갱신
	scope:lia_treaty_nation_1 = {
		add_to_variable_list = {
			name = lia_var_list_treaty_$LIA_EQUALITY_1$
			target = scope:lia_treaty_nation_2
		}
	}
	scope:lia_treaty_nation_2 = {
		add_to_variable_list = {
			name = lia_var_list_treaty_$LIA_EQUALITY_2$
			target = scope:lia_treaty_nation_1
		}
	}
}

	}

		limit = {
			exists = THIS
			THIS != ROOT
			lia_trigger_treaty_check_simple = {
				LIA_NATION_OTHER = ROOT
			}
		}
		#데이터 스캔.
		while = {
			limit = {
				NOT = { has_variable = lia_var_treaty_break }
			}
			lia_effect_treaty_num_puzzle_3n = {	#LIA_eafp_effects.txt의 'lia_effect_treaty_num_puzzle_comp' 참고 바람.
				LIA_PUZZLE = 0

			name = lia_global_var_treaty_num
			value = 1
		}
		ROOT = {
			
			post_notification = lia_message_treaty_violated
		}
	}
	remove_global_variable = lia_global_var_treaty_violate_pass
