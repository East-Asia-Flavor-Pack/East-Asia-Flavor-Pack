bakufu_kaikaku_progress_bar = {
	name = "bakufu_kaikaku_progress_bar_name"
	desc = "bakufu_kaikaku_progress_bar_hitotsubashiha_desc"
	second_desc = "bakufu_kaikaku_progress_bar_nankiha_desc"

	double_sided_bad = yes

	weekly_progress = {
		if = {
			limit = {
				limit = { has_journal_entry = je_bakufu_kaikaku }
			}
			je:je_bakufu_kaikaku = {
				add = {
					#desc = bakufu_kaikaku_progress_bar_balance
					value = {
						value = ROOT.var:hitotsubashiha_var
						divide = {
							value = ROOT.var:hitotsubashiha_var
							add = ROOT.var:nankiha_var
						}
						subtract = "scripted_bar_progress(bakufu_kaikaku_progress_bar)"
					}
				}

				# bakufu_kaikaku_progress_bar_hitotsubashiha_from_character
				add = {
					desc = bakufu_kaikaku_progress_bar_hitotsubashiha_from_character
					value = 0
				}
				# bakufu_kaikaku_progress_bar_hitotsubashiha_from_igs
				add = {
					desc = bakufu_kaikaku_progress_bar_hitotsubashiha_from_igs
					value = {
						value = ig:ig_intelligentsia.ig_clout
						divide = 2
						if = {
							limit = { ig:ig_intelligentsia = { is_in_government = yes } }
							multiply = 6
						}
					}
				}
				add = {
					value = {
						value = ig:ig_intelligentsia.ig_clout
						divide = -2
						if = {
							limit = { ig:ig_intelligentsia = { is_in_government = yes } }
							multiply = 6
						}
					}
				}
				# bakufu_kaikaku_progress_bar_hitotsubashiha_from_building
				add = {
					desc = bakufu_kaikaku_progress_bar_hitotsubashiha_from_building
					value = bakufu_kaikaku_progress_bar_hitotsubashiha_from_building_value
				}
				add = {
					value = bakufu_kaikaku_progress_bar_hitotsubashiha_from_building_value
					multiply = -1
				}
				# bakufu_kaikaku_progress_bar_hitotsubashiha_from_button
				if = {
					limit = {
						has_variable = supporting_hitotsubashiha_var
					}
					add = {
						desc = bakufu_kaikaku_progress_bar_hitotsubashiha_from_button
						value = 0.1
					}
					add = {
						value = -0.1
					}
				}
				# bakufu_kaikaku_progress_bar_hitotsubashiha_from_effect
				add = {
					desc = bakufu_kaikaku_progress_bar_hitotsubashiha_from_effect
					value = 0
				}

				# bakufu_kaikaku_progress_bar_nankiha_from_character
				if = {
					limit = { ruler = { has_template = jap_ienari_tokugawa_template } }
					add = {
						desc = bakufu_kaikaku_progress_bar_nankiha_from_character
						value = {
							value = ruler.popularity
							divide = 50
						}
					}
					add = {
						value = {
							value = ruler.popularity
							divide = -50
						}
					}
				}
				# bakufu_kaikaku_progress_bar_nankiha_from_igs
				add = {
					desc = bakufu_kaikaku_progress_bar_nankiha_from_igs
					value = {
						value = ig:ig_petty_bourgeoisie.ig_clout
						divide = 2
						if = {
							limit = { ig:ig_intelligentsia = { is_in_government = yes } }
							multiply = 6
						}
					}
				}
				add = {
					value = {
						value = ig:ig_petty_bourgeoisie.ig_clout
						divide = -2
						if = {
							limit = { ig:ig_intelligentsia = { is_in_government = yes } }
							multiply = 6
						}
					}
				}
				# bakufu_kaikaku_progress_bar_nankiha_from_building
				#add = {
				#	desc = bakufu_kaikaku_progress_bar_nankiha_from_building
				#	value = bakufu_kaikaku_progress_bar_nankiha_from_building_value
				#}
				#add = {
				#	value = bakufu_kaikaku_progress_bar_nankiha_from_building_value
				#	multiply = -1
				#}
				# bakufu_kaikaku_progress_bar_nankiha_from_button
				if = {
					limit = {
						has_variable = supporting_nankiha_var
					}
					add = {
						desc = bakufu_kaikaku_progress_bar_nankiha_from_button
						value = 0.1
					}
					add = {
						value = -0.1
					}
				}
				# bakufu_kaikaku_progress_bar_nankiha_from_effect
				add = {
					desc = bakufu_kaikaku_progress_bar_nankiha_from_effect
					value = 0
				}
			}
		}
	}

	start_value = 10
	min_value = 0
	max_value = 100
}
