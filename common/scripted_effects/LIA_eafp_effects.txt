		limit = {
			has_role = agitator

		}
		owner = {
			every_scope_state = {
				every_scope_pop = {
					limit = {
						strata = rich
						NOT = { is_pop_type = clergymen }
					}
					change_pop_religion = {
						target = rel:catholic
						value = 0.00003
					}
				}
				every_scope_pop = {
					limit = {
						strata != rich
						NOT = { is_pop_type = clergymen }
					}
					change_pop_religion = {
						target = rel:catholic
						value = 0.00007
					}
				}
			}
		}
	}

lia_effect_cal_catholic_modi_mult = {
	#국가 천주교 신자 비율에 따른 모디파이어 계수 산정 식. 기본형 >> output = 1-{1/(25x+1)^2}, x = religion_percent_country .
	remove_modifier = lia_modi_catholic_plague
	set_variable = {
		name = lia_var_catholic_percent
		value = 0.000	#0.1%단위로 분할
	}
	while = {
		limit = {
			religion_percent_country = {
				target = rel:catholic
				value > var:lia_var_catholic_percent
			}
		}
		change_variable = {
			name = lia_var_catholic_percent
			add = 0.001
		}
	}
	change_variable = {
		name = lia_var_catholic_percent
		multiply = 25
	}
	change_variable = {
		name = lia_var_catholic_percent
		add = 1
	}
	change_variable = {
		name = lia_var_catholic_percent
		multiply = var:lia_var_catholic_percent
	}
	set_variable = {
		name = lia_var_catholic_percent_final
		value = 1
	}
	change_variable = {
		name = lia_var_catholic_percent_final
		divide = var:lia_var_catholic_percent
	}
	change_variable = {
		name = lia_var_catholic_percent_final
		multiply = -1
	}
	change_variable = {
		name = lia_var_catholic_percent_final
		add = 1
	}
	add_modifier = {
		name = lia_modi_catholic_plague
		months = -1
		multiplier = var:lia_var_catholic_percent_final
	}
}
