je_dowager_cixi = { # dowager_cixi_events.1에서 트리거
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	group = je_group_historical_content

	immediate = {
		random_scope_character = {
			limit = { has_template = eafp_chi_xingzhen_yehe_nara_template }
			save_scope_as = Xingzhen_Yehe_Nara_scope
		}
		set_variable = {
			name = cixi_coup
			value = 0.5 # 1이 되면 반란
		}
		# cixi_coup_d는 script_value에서.
	}
	complete = {
		always = no
	}
	on_complete = {
		hidden_effect = {
			remove_variable = cixi_coup
		}
	}
	fail = {
		scope:Xingzhen_Yehe_Nara_scope = {
			is_character_alive = no
			custom_tooltip = {
				text = cixi_not_removed
				has_template = eafp_chi_xingzhen_yehe_nara_template
			}
		}
	}
	on_fail = {
		trigger_event = { id = dowager_cixi_events.3 }
		hidden_effect = {
			remove_variable = cixi_coup
		}
	}
	invalid = {
		custom_tooltip = {
			text = china_shatters_tt
			has_global_variable = china_shatters
		}
	}
	on_invalid = {
		hidden_effect = {
			remove_variable = cixi_coup
		}
	}
	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_dowager_cixi_status
				trigger = {
					always = yes
				}
			}
		}
	}
	on_monthly_pulse = {
		effect = {
			change_variable = {
				name = cixi_coup
				add = cixi_coup_d
			}
			clamp_variable = {
				name = cixi_coup
				max = 1
				min = 0
			}
			if = {
				limit = { var:cixi_coup >= 1 }
				set_variable = {
					name = cixi_coup
					value = 0
				}
				trigger_event = {
					id = dowager_cixi_events.5
					days = { 30 60 }
				}
			}
		}
		random_events = {
			100 = 0
			# TODO
		}
	}
	weight = 1000

	should_be_pinned_by_default = yes

	transferable = no
	can_revolution_inherit = no
}
