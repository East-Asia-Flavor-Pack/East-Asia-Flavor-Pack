japan_figure_effect = {
	clear_global_variable_list = japan_figure_list
	ordered_character_in_void = {
		max = 1000
		order_by = popularity
		check_range_bounds = no

		limit = {
			has_variable = figure_var
		}
		add_to_global_variable_list = {
			name = japan_figure_list
			target = this
		}
	}

}
