donghak_events.7 = {
	type = country_event

	event_image = {
		video = "gfx/event_pictures/southamerica_slaves_night.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/slaves_night"

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	title = donghak_events.7.t
	desc = donghak_events.7.d
	flavor = donghak_events.7.f

        trigger = {
		has_insurrectionary_interest_groups = yes
		ig:ig_rural_folk = {
			is_in_government = no
			is_marginal = no
    		AND = {
	    		ig_approval > -8
	    	}
			NOT = { has_modifier = donghak_sangso_acceptance }
			NOT = { has_modifier = ig_mass_arrests }
			NOT = { has_variable = gobu_rebel }
		}
	}
    

	immediate = {
		set_variable = {
			name =  gobu_rebel
			value = yes
		}
		ig:ig_rural_folk = {
			save_scope_as = ig_rf_on_the_edge
			leader = {
				save_scope_as = union_leader
			}
			}
			ig:ig_landowners = {
			limit = {
				is_in_government = yes
			}
			leader = {
				save_scope_as = landowners_leader
			}
		}
	}

	option = { #진압한다
		default_option = yes
		name = donghak_events.7.a
         add_radicals = {
			pop_type = peasants
			culture = cu:korean
			value = 0.7
		}
			
		ig:ig_rural_folk = {
			
		 join_revolution = yes
		add_modifier = {
	            name = donghak_revolts
	            months = 20
	        }
			add_modifier = {
				name = donghak_gobu_banran
				months = 40
			}
			}
		scope:union_leader = {
				set_ideology = ideology_donghak
			}
		}
	}

	option = {#요구를 받아들인다
		name = donghak_events.7.b
		scope:landowners_leader = {
				kill_character = yes
		}
		ig:ig_landowners = {
			add_modifier = {
				name = donghak_revolts_landowners
				months = 12
			}
		}
	}

}
