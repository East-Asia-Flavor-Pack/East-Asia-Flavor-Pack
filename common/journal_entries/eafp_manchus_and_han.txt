je_manchus_and_han = {
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"
	
	group = je_group_historical_content

	immediate = {
		# 지지도 계산
		set_variable = {
			name = china_manchu_support_var
			value = 0.7
		}
		set_variable = {
			name = china_han_support_var
			value = 0.3
		}
		
		# 권력 계산
		# je_manchus_and_han_pop_pol_str_calc = yes 아직 초기화 안되서 계산이 안됨.
		set_variable = {
			name = manchu_pol_str
			value = 15206493.82
		}
		set_variable = {
			name = han_pol_str
			value = 19954764.01
		}
		set_variable = {
			name = china_manchu_power_from_pol_str_var
			value = {
				value = var:manchu_pol_str
				divide = {
					value = var:manchu_pol_str
					add = var:han_pol_str
				}
			}
		}
		set_variable = {
			name = china_han_power_from_pol_str_var
			value = {
				value = 1
				subtract = var:china_manchu_power_from_pol_str_var
			}
		}
		remove_variable = manchu_pol_str
		remove_variable = han_pol_str

		set_variable = {
			name = china_manchu_power_var
			value = var:china_manchu_power_from_pol_str_var
		}
		set_variable = {
			name = china_han_power_var
			value = var:china_han_power_from_pol_str_var
		}
		je_manchus_and_han_modifier_calc = yes
	}

	invalid = {
		custom_tooltip = {
			text = china_shatters_tt
			has_global_variable = china_shatters
		}
	}

	on_invalid = {
		remove_variable = china_manchu_support_var
		remove_variable = china_han_support_var
		remove_variable = china_manchu_power_var
		remove_variable = china_han_power_var
		remove_variable = china_manchu_power_from_pol_str_var
		remove_variable = china_han_power_from_pol_str_var
		remove_variable = china_manchu_multiplier_var
		remove_variable = china_han_multiplier_var
		if = {
			limit = { has_modifier = manchu_influence1 }
			remove_modifier = manchu_influence1
		}
		if = {
			limit = { has_modifier = manchu_influence2 }
			remove_modifier = manchu_influence2
		}
		if = {
			limit = { has_modifier = han_influence1 }
			remove_modifier = han_influence1
		}
		if = {
			limit = { has_modifier = han_influence2 }
			remove_modifier = han_influence2
		}
	}

	complete = {
		NOT = { has_law = law_type:law_racial_segregation }
	}

	on_complete = {
		remove_variable = china_manchu_support_var
		remove_variable = china_han_support_var
		remove_variable = china_manchu_power_var
		remove_variable = china_han_power_var
		remove_variable = china_manchu_power_from_pol_str_var
		remove_variable = china_han_power_from_pol_str_var
		remove_variable = china_manchu_multiplier_var
		remove_variable = china_han_multiplier_var
		if = {
			limit = { has_modifier = manchu_influence1 }
			remove_modifier = manchu_influence1
		}
		if = {
			limit = { has_modifier = manchu_influence2 }
			remove_modifier = manchu_influence2
		}
		if = {
			limit = { has_modifier = han_influence1 }
			remove_modifier = han_influence1
		}
		if = {
			limit = { has_modifier = han_influence2 }
			remove_modifier = han_influence2
		}
	}

	modifiers_while_active = {
		manchus_above_all
	}
	
	on_monthly_pulse = {
		effect = {
			# 지지도 계산
			
			# 권력 계산
			je_manchus_and_han_pop_pol_str_calc = yes
			set_variable = {
				name = china_manchu_power_from_pol_str_var
				value = {
					value = var:manchu_pol_str
					divide = {
						value = var:manchu_pol_str
						add = var:han_pol_str
					}
				}
			}
			set_variable = {
				name = china_han_power_from_pol_str_var
				value = {
					value = 1
					subtract = var:china_manchu_power_from_pol_str_var
				}
			}
			remove_variable = manchu_pol_str
			remove_variable = han_pol_str

			set_variable = {
				name = china_manchu_power_var
				value = var:china_manchu_power_from_pol_str_var
			}
			set_variable = {
				name = china_han_power_var
				value = var:china_han_power_from_pol_str_var
			}
			je_manchus_and_han_modifier_calc = yes
		}
	}

	on_yearly_pulse = {
		effect = {
			if = {
				limit = {
					has_technology_researched = nationalism
					NOT = { has_variable = manhan1 }
				}
				set_variable = manhan1
				trigger_event = {
					id = manchus_and_han_events.1
					months = { 0 12 }
				}
			}
			if = {
				limit = {
					has_technology_researched = pan-nationalism
					NOT = { has_variable = manhan2 }
				}
				set_variable = manhan2
				trigger_event = {
					id = manchus_and_han_events.2
					months = { 12 60 }
				}
			}
			if = {
				limit = { has_variable = manhan1 }
				random_list = {
					40 = {}
					5 = {
						modifier = {
							if = {
								limit = { cu:han = { culture_is_discriminated_in = ROOT } }
								multiply = 2
							}
						}
						modifier = {
							if = {
								limit = { has_variable = manhan2 }
								multiply = 2
							}
						}
						modifier = {
							if = {
								limit = { has_law = law_type:law_landed_voting }
								multiply = 0.9
							}
						}
						modifier = {
							if = {
								limit = { has_law = law_type:law_census_voting }
								multiply = 0.8
							}
						}
						modifier = {
							if = {
								limit = { has_law = law_type:law_wealth_voting }
								multiply = 0.7
							}
						}
						modifier = {
							if = {
								limit = { has_law = law_type:law_universal_suffrage }
								multiply = 0.6
							}
						}
						trigger_event = {
							id = manchus_and_han_events.3
							months = { 0 12 }
						}
					}
				}
			}
		}
	}

	progress_desc = je_manchus_and_han_progress

	status_desc = je_manchus_and_han_status

	weight = 1000
	should_be_pinned_by_default = no
}
