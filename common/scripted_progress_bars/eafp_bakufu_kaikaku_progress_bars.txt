bakufu_kaikaku_progress_bar = {
	name = "bakufu_kaikaku_progress_bar_name"
	desc = "bakufu_kaikaku_progress_bar_hitotsubashiha_desc"
	second_desc = "bakufu_kaikaku_progress_bar_nankiha_desc"

	double_sided_bad = yes

	weekly_progress = {
		if = {
			limit = {
				limit = { has_journal_entry = je_bakufu_kaikaku }
			}
			je:je_bakufu_kaikaku = { # 실제 계산은 scripted effect로
				# bakufu_kaikaku_progress_bar_hitotsubashiha_from_character
				add = {
					desc = bakufu_kaikaku_progress_bar_hitotsubashiha_from_character
					value = var:bakufu_kaikaku_progress_bar_hitotsubashiha_from_character
				}
				add = {
					value = var:bakufu_kaikaku_progress_bar_hitotsubashiha_from_character
					multiply = -1
				}
				# bakufu_kaikaku_progress_bar_hitotsubashiha_from_igs
				add = {
					desc = bakufu_kaikaku_progress_bar_hitotsubashiha_from_igs
					value = var:bakufu_kaikaku_progress_bar_hitotsubashiha_from_igs
				}
				add = {
					value = var:bakufu_kaikaku_progress_bar_hitotsubashiha_from_igs
					multiply = -1
				}
				# bakufu_kaikaku_progress_bar_hitotsubashiha_from_button
				if = {
					limit = {
						has_variable = supporting_hitotsubashiha_var
					}
					add = {
						desc = bakufu_kaikaku_progress_bar_hitotsubashiha_from_button
						value = 0.1
					}
					add = {
						value = -0.1
					}
				}
				# bakufu_kaikaku_progress_bar_hitotsubashiha_from_modifier
				add = {
					desc = bakufu_kaikaku_progress_bar_hitotsubashiha_from_modifier
					value = 0
				}

				# bakufu_kaikaku_progress_bar_nankiha_from_character
				add = {
					desc = bakufu_kaikaku_progress_bar_nankiha_from_character
					value = var:bakufu_kaikaku_progress_bar_nankiha_from_character
				}
				add = {
					value = var:bakufu_kaikaku_progress_bar_nankiha_from_character
					multiply = -1
				}
				# bakufu_kaikaku_progress_bar_nankiha_from_igs
				add = {
					desc = bakufu_kaikaku_progress_bar_nankiha_from_igs
					value = var:bakufu_kaikaku_progress_bar_nankiha_from_igs
				}
				add = {
					value = var:bakufu_kaikaku_progress_bar_nankiha_from_igs
					multiply = -1
				}
				# bakufu_kaikaku_progress_bar_nankiha_from_button
				if = {
					limit = {
						has_variable = supporting_nankiha_var
					}
					add = {
						desc = bakufu_kaikaku_progress_bar_nankiha_from_button
						value = 0.1
					}
					add = {
						value = -0.1
					}
				}
				# bakufu_kaikaku_progress_bar_nankiha_from_modifier
				add = {
					desc = bakufu_kaikaku_progress_bar_nankiha_from_modifier
					value = 0
				}

				add = { # 밸런스
					value = {
						value = ROOT.var:hitotsubashiha_var
						divide = {
							value = ROOT.var:hitotsubashiha_var
							add = ROOT.var:nankiha_var
						}
						subtract = "scripted_bar_progress(bakufu_kaikaku_progress_bar)"
					}
				}
			}
		}
	}

	start_value = 10
	min_value = 0
	max_value = 100
}
