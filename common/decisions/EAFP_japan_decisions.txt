

sonno_kanri_decision = {

		exists = c:NIP
		c:NIP = THIS
		has_global_variable = kodokanki_var

		exists = c:NIP

			name = sonno_kanri_cooltime
			months = 36
		}

            id = before_meiji.300
        }
	}
	ai_chance = {
		base = 100
    }
