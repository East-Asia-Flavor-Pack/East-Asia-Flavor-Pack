country_owns_entire_strategic_region_gui = {
	saved_scopes = {
		country
    }

	is_valid = {
		any_scope_state = {
			count = all
			owner = scope:country
		}
	}
}

country_or_subject_owns_entire_strategic_region_gui = {
	saved_scopes = {
		country
    }

	is_valid = {
		any_scope_state = {
			count = all
			owner = {
				OR = { # 설마 이 이상이 필요할까...
					THIS = scope:country
					overlord ?= scope:country
					top_overlord ?= scope:country
				}
			}
		}
	}
}
