button_je_eafp_manchuria_settlement = {
	name = button_je_eafp_manchuria_settlement
	desc = button_je_eafp_manchuria_settlement_desc

	visible = {
		c:KOR ?= this
		is_involved_in_journal_entry = je_eafp_manchurian_management
		NOT = { eafp_manchuria_settlement_ready = yes }
	}

	cooldown = { days = 730 }

	possible = {
		eafp_manchuria_owns_all_manchurian_states = yes
		any_scope_state = {
			owner = ROOT
			is_in_geographic_region = geographic_region_korea
			any_scope_pop = {
				has_pop_culture = korean
				OR = {
					is_pop_type = peasants
					is_pop_type = laborers
				}
			}
		}
	}

	effect = {
		random_scope_state = {
			limit = {
				owner = ROOT
				is_in_geographic_region = geographic_region_korea
				any_scope_pop = {
					has_pop_culture = korean
					OR = {
						is_pop_type = peasants
						is_pop_type = laborers
					}
				}
			}
			random_scope_pop = {
				limit = {
					has_pop_culture = korean
					OR = {
						is_pop_type = peasants
						is_pop_type = laborers
					}
				}
				save_scope_as = eafp_manchuria_migrant_pop
			}
		}
		random_scope_state = {
			limit = {
				eafp_manchuria_is_core_manchurian_state = yes
				owner = ROOT
			}
			save_scope_as = eafp_manchuria_migrant_target_state
			add_cultural_community = cu:korean
			add_modifier = {
				name = eafp_manchurian_settlement_state
				days = normal_modifier_time
			}
		}
		scope:eafp_manchuria_migrant_pop = {
			move_pop = scope:eafp_manchuria_migrant_target_state
		}
	}

	ai_chance = {
		value = 20
	}
}

button_je_eafp_manchuria_expand_arable_land = {
	name = button_je_eafp_manchuria_expand_arable_land
	desc = button_je_eafp_manchuria_expand_arable_land_desc

	visible = {
		c:KOR ?= this
		is_involved_in_journal_entry = je_eafp_manchurian_management
	}

	cooldown = { days = 1095 }

	possible = {
		eafp_manchuria_owns_all_manchurian_states = yes
	}

	effect = {
		every_scope_state = {
			limit = {
				eafp_manchuria_is_core_manchurian_state = yes
				owner = ROOT
			}
			state_region = {
				add_arable_land = 5
			}
			add_modifier = {
				name = eafp_manchurian_settlement_state
				days = normal_modifier_time
			}
		}
	}

	ai_chance = {
		value = 15
	}
}
