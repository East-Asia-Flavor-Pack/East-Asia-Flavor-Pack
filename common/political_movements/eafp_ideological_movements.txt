eafp_movement_rel_catholic = {
	category = eafp_movement_category_propagation
	#category = movement_category_religious
	
	ideology = ideology_eafp_catholic_movement
	
	character_ideologies = {
		ideology_eafp_missionary
	}
	
	creation_trigger = {
		NOT = { country_has_state_religion = rel:catholic }	#활약국의 국교가 캣홀릭이면 의미가 없음
		NOT = { has_law = law_type:law_freedom_of_conscience }	#활약국이 이미 신앙의 자유가 있다면 의미가 없음
		NOT = { has_law = law_type:law_total_separation }
	}	
	creation_weight = {
		value = 200
	}	
	on_created = {}

	disband_trigger = {
		rel:catholic = {
			is_state_religion = root
		}
	}
	
	#일단 보존처리!
	#religion_selection_trigger = {
	#	NOT = { is_state_religion = scope:country }
	#	NOT = { rel:atheist = this }
	#	rel:catholic = this
	#	#유학 저널 클리어시 보상으로 포교운동을 꺼버리는것을 고려해봐.
	#}
	#religion_selection_weight = {
	#	value = 200
	#}
	
	character_support_trigger = {
		religion = rel:catholic
		#OR = {
		#	is_interest_group_type = ig_devout
		#	has_ideology = ideology:ideology_traditionalist
		#	has_ideology = ideology:ideology_theocrat
		#	has_ideology = ideology:ideology_integralist
		#	has_ideology = ideology:ideology_eafp_missionary
		#}		
	}
	character_support_weight = {
		value = 100
		if = {
			limit = {
				interest_group ?= {
					is_interest_group_type = ig_devout
				}
			}
			multiply = {
				value = 5
			}
		}
		if = {
			limit = {
				has_ideology = ideology:ideology_eafp_missionary
			}
			add = {
				value = 100
			}
		}
		if = {
			limit = {
				has_ideology = ideology:ideology_eafp_missionary
			}
			multiply = {
				value = 10
			}
		}
	}
	
	pop_support_trigger = {
		religion = rel:catholic
	}

	pop_support_factors = {
		movement_support_poverty
		movement_support_low_literacy
		movement_support_religious_discrimination
		movement_support_clergymen
		movement_support_aristocrats
		movement_support_peasants
		movement_support_corporatism_tech
		movement_support_eafp_catholic
	}

	pop_support_weight = {
		add = {
			value = 10
			desc = "POP_BASE_SUPPORT"
		}	
		
		if = {
			limit = { 
				religion = rel:catholic
			}
			
			add = {
				value = 25
				desc = "EAFP_POP_CATHOLIC"
			}
		}
		
		if = {
			limit = { 
				standard_of_living < 15
			}
			
			add = {
				value =	standard_of_living
				subtract = 15
				multiply = -5
				desc = "POP_POVERTY"
			}
		}	

		if = {
			limit = {
				is_pop_type = clergymen
			}
			add = {
				value = 45
				desc = "POP_CLERGYMEN"
			}
		}
		else_if = {
			limit = {
				is_pop_type = aristocrats
			}
			add = {
				value = 10
				desc = "POP_ARISTOCRATS"
			}
		}	
		else_if = {
			limit = {
				is_pop_type = peasants
			}
			add = {
				value = 10
				desc = "POP_PEASANTS"
			}
		}	

		add = { # Weakened by increasing literacy rates
			desc = "POP_LITERACY"
			this = {
				value = literacy_rate
				multiply = -2
			}
		}
		
		if = {
			limit = {
				state = {
					state_religious_acceptance = {  
						target = PREV.religion
						value < religious_acceptance_high
					}
				}
			}
			multiply = {
				desc = "POP_RELIGIOUS_DISCRIMINATION"
				value = 100
				subtract = pop_acceptance
				divide = 25 # Maxes out at 4x at 0% acceptance, starts being a liability at 75% acceptance
			}		
		}
		
		if = {
			limit = {
				owner ?= {
					has_technology_researched = corporatism
				}
			}
			multiply = {
				value = 1.5
				desc = "CORPORATISM_TECH"
			}
		}

		# Favor cultural movement if necessary
		if = {
			limit = {
				owner = { 
					any_political_movement = {
						is_political_movement_type = movement_cultural_minority
						this.culture ?= root.culture
					} 
				}
			}
			if = {
				limit = {
					owner = { cultural_acceptance_base = { target = culture value <= acceptance_status_1 } }
				}
				multiply = 0.1
			}
			else_if = {
				limit = {
					owner = { cultural_acceptance_base = { target = culture value <= acceptance_status_2 } }
				}
				multiply = 0.25	
			}
			else_if = {
				limit = {
					owner = { cultural_acceptance_base = { target = culture value <= acceptance_status_3 } }
				}
				multiply = 0.5	
			}
		}	
	}	
	
	secession = {
		possible = {
			political_movement_popular_support >= 0.25
			owner ?= {
				any_scope_state = {
					religion_percent_state = {
						target = scope:political_movement.religion
						value >= 0.10
					}
				}
			}
		}
	
		weight = {
			value = 1000
		}	
		
		state_weight = {
			value = 0
		
			if = {
				limit = { 
					is_capital = no
				}

				if = {
					limit = {
						religion_percent_state = {
							target = scope:political_movement.religion
							value >= 0.75
						}
					}
					add = 250
				}
				else_if = {
					limit = {
						religion_percent_state = {
							target = scope:political_movement.religion
							value >= 0.5
						}
					}
					add = 150
				}
				else_if = {
					limit = {
						religion_percent_state = {
							target = scope:political_movement.religion
							value >= 0.25
						}
					}
					add = 50
				}
				else_if = {
					limit = {
						religion_percent_state = {
							target = scope:political_movement.religion
							value >= 0.1
						}
					}
					add = 10
				}
			}			
		}	

		target_fraction_of_states = {
			value = political_movement_popular_support
			add = scope:clout
			multiply = 3
			min = 0.15
			max = 0.75
		}		
	}
	
	additional_radicalism_factors = {
		add = {
			value = "owner.average_sol"
			subtract = "owner.average_sol_for_religion(scope:religion)"
			multiply = 0.05
			max = 0.5
			min = 0
			desc = "STANDARD_OF_LIVING_FOR_RELIGION"
		}
		add = {
			value = -0.2
			desc = "EAFP_FACTOR_SECRET_COMMUNITY"
		}
	}
}

eafp_movement_sonno_joi = {
	category = movement_category_ideological

	ideology = ideology_sonno_joi_movement
	character_ideologies = {
		ideology_sonno_joi
		ideology_modernizer_leader
		ideology_protectionist
	}

	creation_trigger = {
        c:JAP ?= this
		has_law = law_type:law_bakufu
	}	

	creation_weight = {
		value = 100
	}

	on_created = {
		add_movement_enthusiasm_modifier = yes
	}

	character_support_trigger = {
		culture = cu:japanese
		OR = {
			has_ideology = ideology:ideology_sonno_joi
			has_ideology = ideology:ideology_modernizer_leader
			has_ideology = ideology:ideology_protectionist
		}
	}
	character_support_weight = {
		value = 100
		if = {
			limit = {
				has_ideology = ideology:ideology_sonno_joi
			}
			multiply = {
				value = 5
			}
		}
		if = {
			limit = {
				OR = {
					is_interest_group_type = ig_devout
					is_interest_group_type = ig_intelligentsia
					is_interest_group_type = ig_rural_folk
				}
			}
			multiply = {
				value = 2
			}
		}
	}

	can_pressure_interest_group = {
		NOT = { is_interest_group_type = ig_landowners }
		NOT = { is_interest_group_type = ig_armed_forces }
	}

	pop_support_trigger = {
		culture = cu:japanese
	}

	pop_support_factors = {
		movement_support_high_literacy
		movement_support_below_expected_sol
		movement_support_radicalization
		movement_support_clergymen
		movement_support_farmers
		movement_support_academics
		movement_support_bureaucrats
		movement_support_clerks
		movement_support_je_bakuhantaisei
	}

	pop_support_weight = {
		if = {
			limit = {
				exists = scope:political_movement
				scope:political_movement ?= {
					OR = {
						is_being_suppressed = yes
						is_being_bolstered = yes
					}
				}
			}
			if = {
				limit = {
					scope:political_movement = {
						is_being_suppressed = yes
					}
				}
				add = {
					value = suppression_base_value
					multiply = {
						value = -0.5
						add = owner.modifier:country_suppression_attraction_factor
					}
					desc = "concept_suppressed"
				}
			}
			else = {
				add = {
					value = bolstering_base_value
					multiply = {
						value = 0.5
						add = owner.modifier:country_bolster_attraction_factor
					}
					desc = "concept_bolstered"
				}
			}
		}

		if = {
			limit = {
				is_pop_type = clergymen
			}
			add = {
				value = 20
				desc = "POP_CLERGYMEN"
			}
		}
		else_if = {
			limit = {
				is_pop_type = farmers
			}
			add = {
				value = 20
				desc = "POP_FARMERS"
			}
		}
		else_if = {
			limit = {
				is_pop_type = academics
			}
			add = {
				value = 30
				desc = "POP_ACADEMICS"
			}
		}
		else_if = {
			limit = {
				is_pop_type = bureaucrats
			}
			add = {
				value = 20
				desc = "POP_BUREAUCRATS"
			}
		}
		else_if = {
			limit = {
				is_pop_type = clerks
			}
			add = {
				value = 20
				desc = "POP_CLERKS"
			}
		}
		else = {
			add = {
				value = 10
				desc = "WAR_EXHAUSTION_BASE"
			}
		}

		if = {
			limit = {
				standard_of_living < state.average_expected_sol # It is intentional that a labourer being well-off doesn't decrease support for this.
			}
			add = { # Increases likelihood of support if impoverished.
				desc = "IS_BELOW_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
				max = 10
			}
		}

		multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.10
		}

		multiply = { # Multiplied by 2x max
			desc = "POP_RADICALS"
			value = pop_radical_fraction
			add = 1
		}

		if = { # 3배까지 ㄱㄴ
			limit = { owner ?= { has_journal_entry = je_bakuhantaisei } }
			owner ?= {
				multiply = {
					desc = "from_bakuhantaisei_je"
					value = 4000
					je:je_bakuhantaisei ?= {
						subtract = "scripted_bar_progress(bakuhantaisei_bakufu_authority_progress_bar)"
					}
					divide = 2000
					add = 1
				}
			}
		}
	}

	revolution = {
		possible = {
			political_movement_support >= 0.05
		}

		weight = {
			value = 100
		}

		state_weight = {
			value = 0
			if = {
				limit = {
					s:STATE_TOHOKU.region_state:JAP ?= THIS
					owner ?= { has_journal_entry = je:je_bakuhantaisei_TOHOKU }
				}
				owner = {
					add = {
						value = 100
						subtract = bakuhantaisei_TOHOKU_loyalty_progress_bar_value
						add = bakuhantaisei_TOHOKU_independency_progress_bar_value
					}
				}
			}
			if = {
				limit = {
					s:STATE_KANTO.region_state:JAP ?= THIS
					owner ?= { has_journal_entry = je:je_bakuhantaisei_KANTO }
				}
				owner = {
					add = {
						value = 100
						subtract = bakuhantaisei_KANTO_loyalty_progress_bar_value
						add = bakuhantaisei_KANTO_independency_progress_bar_value
					}
				}
			}
			if = {
				limit = {
					s:STATE_CHUBU.region_state:JAP ?= THIS
					owner ?= { has_journal_entry = je:je_bakuhantaisei_CHUBU }
				}
				owner = {
					add = {
						value = 100
						subtract = bakuhantaisei_CHUBU_loyalty_progress_bar_value
						add = bakuhantaisei_CHUBU_independency_progress_bar_value
					}
				}
			}
			if = {
				limit = {
					s:STATE_KANSAI.region_state:JAP ?= THIS
					owner ?= { has_journal_entry = je:je_bakuhantaisei_KANSAI }
				}
				owner = {
					add = {
						value = 100
						subtract = bakuhantaisei_KANSAI_loyalty_progress_bar_value
						add = bakuhantaisei_KANSAI_independency_progress_bar_value
					}
				}
			}
			if = {
				limit = {
					s:STATE_KYUSHU.region_state:JAP ?= THIS
					owner ?= { has_journal_entry = je:je_bakuhantaisei_KYUSHU }
				}
				owner = {
					add = {
						value = 100
						subtract = bakuhantaisei_KYUSHU_loyalty_progress_bar_value
						add = bakuhantaisei_KYUSHU_independency_progress_bar_value
					}
				}
			}
