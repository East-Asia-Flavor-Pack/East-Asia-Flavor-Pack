		}

						NOT = { owner = ROOT }

						hidden_trigger = {
							NOT = { has_variable_list = chi_war_dynamism_varlist }
						}
						owner = {
							has_war_with = ROOT
							OR = {
								AND = {

							}
						}
					}

						name = chi_war_dynamism_modifier
