je_taiping_house_on_sand = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_internal_affairs

	immediate = {
		c:CHI = {
			save_scope_as = china_scope
		}
		if = {
			limit = {
				NOT = { has_variable = taiping_internal_conflict_resolved }
				NOT = { has_journal_entry = je_taiping_internal_conflict }
			}
			add_journal_entry = { type = je_taiping_internal_conflict }
		}
	}

	complete = {
		OR = {
			custom_tooltip = {
				text = taiping_house_on_sand_complete
				hidden_trigger = {
					NOT = { exists = c:CHI }
				}
			}
			custom_tooltip = {
				text = china_shatters_tt
				has_global_variable = china_shatters
			}
		}
	}

	invalid = {
		NOT = { has_journal_entry = je_taiping_revolution }
	}

	on_complete = {
		hidden_effect = {
			set_variable = taiping_house_on_sand_succeeded
			eafp_taiping_cleanup_administration = yes
		}
		trigger_event = { id = heavenly.1005 popup = yes }
	}

	on_invalid = {
		hidden_effect = {
			set_variable = taiping_house_on_sand_ended
			eafp_taiping_cleanup_administration = yes
		}
	}

	fail = {
		custom_tooltip = {
			text = taiping_house_on_sand_fail
			scope:journal_entry = {
				"scripted_bar_progress(taiping_revolutionary_fervor_progress_bar)" <= 0
			}
		}
	}

	on_fail = {
		set_variable = taiping_house_on_sand_failed
		set_variable = taiping_house_on_sand_ended
		hidden_effect = {
			eafp_taiping_cleanup_administration = yes
		}
		trigger_event = { id = heavenly.1004 popup = yes }
	}

	on_monthly_pulse = {
		effect = {
			eafp_taiping_house_on_sand_monthly = yes
		}
	}

	scripted_progress_bar = taiping_revolutionary_fervor_progress_bar
	progress_desc = je_taiping_house_on_sand_progress

	event_outcome_completed_effect_desc = {
		header = heavenly.1005.a
		effect = {
			add_modifier = {
				name = tianming
				years = 5
			}
		}
	}

	weight = 1000
	should_be_pinned_by_default_uninvolved_or_context = yes

	transferable = no
}

je_taiping_internal_conflict = {
	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	group = je_group_internal_affairs

	immediate = {
		c:CHI = {
			save_scope_as = china_scope
		}
		if = {
			limit = { NOT = { has_variable = taiping_internal_conflict } }
			set_variable = {
				name = taiping_internal_conflict
				value = 10
			}
		}
		set_bar_progress = {
			name = taiping_internal_conflict_progress_bar
			value = var:taiping_internal_conflict
		}
	}

	complete = {
		custom_tooltip = {
			text = taiping_internal_conflict_complete
			hidden_trigger = {
				has_variable = taiping_house_on_sand_succeeded
			}
		}
	}

	on_complete = {
		remove_modifier = internal_conflict_in_tpg
		remove_modifier = taiping_coup_aftermath_modifier
		remove_variable = taiping_internal_conflict
		remove_variable = taiping_house_on_sand_succeeded
		remove_variable = taiping_house_on_sand_ended
	}

	invalid = {
		OR = {
			has_variable = taiping_house_on_sand_ended
			NOR = {
				has_law_or_variant = law_type:law_monarchy
				has_law_or_variant = law_type:law_theocracy
			}
		}
	}

	on_invalid = {
		remove_modifier = taiping_coup_aftermath_modifier
		remove_variable = taiping_internal_conflict
		remove_variable = taiping_house_on_sand_succeeded
		remove_variable = taiping_house_on_sand_ended
	}

	fail = {
		custom_tooltip = {
			text = taiping_internal_conflict_fail
			scope:journal_entry = { "scripted_bar_progress(taiping_internal_conflict_progress_bar)" >= 100 }
		}
	}

	on_fail = {
		set_variable = {
			name = taiping_internal_conflict
			value = 100
		}
		trigger_event = { id = heavenly.1008 popup = yes }
	}

	on_monthly_pulse = {
		effect = {
			eafp_taiping_internal_conflict_monthly = yes
		}
	}

	scripted_progress_bar = taiping_internal_conflict_progress_bar

	weight = 1000
	should_be_pinned_by_default_uninvolved_or_context = yes

	transferable = no
}
