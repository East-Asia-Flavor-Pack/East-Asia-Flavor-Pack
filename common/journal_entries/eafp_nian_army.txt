je_nian_army = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_historical_content

	is_shown_when_inactive = {
		ROOT = c:CHI
		is_player = yes
	}

	possible = {
		any_scope_state = {
			region = sr:region_north_china
			state_is_in_china_proper = yes
			devastation >= 0.25
		}
	}

	immediate = {
		set_variable = {
			name = nian_army_progress_var
			value = 0
		}
		random_scope_state = {
			limit = { region = sr:region_north_china }
			land_hq = {
				save_scope_as = north_china_hq
			}
		}

		set_variable = {
			name = nian_army_progress_var_d_mobilized_garrison
			value = {
				value = scope:north_china_hq.num_garrison_units
				multiply = -0.02
			}
		}
		set_variable = {
			name = nian_army_progress_var_d_stationing_garrison
			value = 0
		}
		scope:north_china_hq = {
			every_military_formation = {
				limit = {
					is_army = yes
					is_on_front = no
				}
				every_combat_unit = {
					ROOT = {
						change_variable = {
							name = nian_army_progress_var_d_stationing_garrison
							add = 1
						}
					}
				}
			}
		}
		change_variable = {
			name = nian_army_progress_var_d_stationing_garrison
			multiply = -0.005
		}
		set_variable = {
			name = nian_army_progress_var_d
			value = {
				value = 2
				add = var:nian_army_progress_var_d_mobilized_garrison
				add = var:nian_army_progress_var_d_stationing_garrison
			}
		}
		c:TPG = {
			save_scope_as = tpg_scope
		}
		trigger_event = { id = nian_army.1 popup = yes }
		trigger_event = { id = nian_army.4 days = 1 }
	}

	complete = {
		custom_tooltip = {
			text = je_nian_army_complete
			exists = var:nian_army_progress_var
			var:nian_army_progress_var < 0
		}
	}

	on_complete = {
		remove_variable = nian_army_progress_var
		remove_variable = nian_army_progress_var_d
		trigger_event = { id = nian_army.2 popup = yes }
	}

	fail = {
		custom_tooltip = {
			text = je_nian_army_fail
			exists = var:nian_army_progress_var
			var:nian_army_progress_var >= 100
		}
	}

	on_fail = {
		remove_variable = nian_army_progress_var
		remove_variable = nian_army_progress_var_d
		trigger_event = { id = nian_army.3 popup = yes }
	}

	invalid = {
		custom_tooltip = {
			text = china_shatters_tt
			has_global_variable = china_shatters
		}
	}

	on_invalid = {
		remove_variable = nian_army_progress_var
	}

	modifiers_while_active = {
		
	}

	on_weekly_pulse = {
		effect = {
			set_variable = {
				name = nian_army_progress_var_d_mobilized_garrison
				value = {
					value = scope:north_china_hq.num_garrison_units
					multiply = -0.02
				}
			}
			set_variable = {
				name = nian_army_progress_var_d_stationing_garrison
				value = 0
			}
			scope:north_china_hq = {
				every_military_formation = {
					limit = {
						is_army = yes
						is_on_front = no
					}
					every_combat_unit = {
						ROOT = {
							change_variable = {
								name = nian_army_progress_var_d_stationing_garrison
								add = 1
							}
						}
					}
				}
			}
			change_variable = {
				name = nian_army_progress_var_d_stationing_garrison
				multiply = -0.005
			}
			set_variable = {
				name = nian_army_progress_var_d
				value = {
					value = 2
					add = var:nian_army_progress_var_d_mobilized_garrison
					add = var:nian_army_progress_var_d_stationing_garrison
				}
			}
		}
	}

	on_monthly_pulse = {
		effect = {			
			set_variable = {
				name = nian_army_progress_var_d_mobilized_garrison
				value = {
					value = scope:north_china_hq.num_garrison_units
					multiply = -0.02
				}
			}
			set_variable = {
				name = nian_army_progress_var_d_stationing_garrison
				value = 0
			}
			scope:north_china_hq = {
				every_military_formation = {
					limit = {
						is_army = yes
						is_on_front = no
					}
					every_combat_unit = {
						ROOT = {
							change_variable = {
								name = nian_army_progress_var_d_stationing_garrison
								add = 1
							}
						}
					}
				}
			}
			change_variable = {
				name = nian_army_progress_var_d_stationing_garrison
				multiply = -0.005
			}
			set_variable = {
				name = nian_army_progress_var_d
				value = {
					value = 2
					add = var:nian_army_progress_var_d_mobilized_garrison
					add = var:nian_army_progress_var_d_stationing_garrison
				}
			}
			change_variable = {
				name = nian_army_progress_var
				add = var:nian_army_progress_var_d
			}
		}
		random_events = {
			10 = nian_army.11
			10 = nian_army.12
			10 = nian_army.13
			10 = nian_army.14
			10 = nian_army.15
			10 = nian_army.16
			10 = nian_army.17
			10 = nian_army.18
			10 = nian_army.19
			50 = 0
		}
	}

	should_be_pinned_by_default = yes

	weight = 8000

	current_value = {
		value = var:nian_army_progress_var
	}

	goal_add_value = {
		add = 100
	}

	# status_desc = je_nian_army_desc

	progressbar = yes
}
