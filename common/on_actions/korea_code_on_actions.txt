korea_on_monthly_pulse_country = {
	effect = {
		if = {
			limit = {
				c:KOR ?= this
			}
			set_global_variable = {
				name = lia_global_val_stress_cost
				value = this.GDP
			}
			change_global_variable = {
				name = lia_global_val_stress_cost
				divide = 100000
			}
			change_global_variable = {
				name = lia_global_val_stress_cost
				multiply = -40
			}
			change_global_variable = {
				name = lia_global_val_stress_cost
				add = -2000
			}
		}
	}
}

korea_on_revolution_start = {
	effect = {
		scope:target ?= {
			if = {
				limit = {
					c:KOR ?= PREV
					has_law = law_type:law_monarchy
				}
				ordered_interest_group = {
					limit = {
						is_in_government = yes
					}
					order_by = ig_clout
					save_scope_as = lia_revolt_kings_ig
				}
				if = {
					limit = {
						scope:lia_revolt_kings_ig = {
							OR = {
								is_interest_group_type = ig_landowners
								is_interest_group_type = ig_devout
							}
						}
						PREV = {
							ruler = {
								has_variable = is_yihwan_eafp
							}
							any_scope_character = {
								NOT = { has_variable = is_yihui_eafp }
								count = all
							}
						}
					}
					create_character = {	#익평군 이희 추대
						first_name = "Hui2"
						last_name = "Yi"
						historical = yes
						ruler = yes
						birth_date = 1824.10.16
						interest_group = scope:lia_revolt_kings_ig
						on_created = {
							add_random_trait = skill
							set_variable = {
								name = is_yihui_eafp
							}
							add_modifier = {
								name = eafp_modi_revolt_king
								months = 12
								is_decaying = yes
							}
							add_modifier = {
								name = eafp_modi_kings_health_bad
								months = -1
							}
							set_variable = is_kor_revolt_king_eafp
						}
					}
					PREV = {
						set_variable = lia_var_yihui_unavailable
					}
				}
				else = {
					create_character = {	#이씨 추대
						last_name = "Yi"
						ruler = yes
						age = { 10 32 }
						interest_group = scope:lia_revolt_kings_ig
						on_created = {
							add_random_trait = skill
							add_modifier = {
								name = eafp_modi_revolt_king
								months = 60
								is_decaying = yes
							}
							set_variable = is_kor_revolt_king_eafp
						}
					}
				}
			}
		}
		#
		
	}
}
