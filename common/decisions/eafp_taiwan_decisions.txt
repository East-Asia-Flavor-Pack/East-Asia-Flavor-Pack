	is_shown = {
		c:KOR ?= THIS
		is_involved_in_journal_entry = je_eafp_taiwan_pioneer

		NOT = { has_variable = eafp_taiwan_pioneer_failed_var }
		NOT = { has_variable = eafp_taiwan_pioneer_resolved_var }
		NOT = { has_variable = eafp_taiwan_survey_cooldown_var }
		var:eafp_taiwan_pioneer_var < 100
	}

	possible = {
		is_at_war = no
		NOT = { has_variable = on_expedition }
		NOT = { has_variable = eafp_taiwan_survey_active_var }
		c:RYU ?= {
			is_subject_of = ROOT
		}

	}

	when_taken = {
		set_variable = on_expedition
		set_variable = eafp_taiwan_survey_active_var
		set_variable = { name = eafp_taiwan_survey_cooldown_var value = yes days = 1095 }
		trigger_event = { id = eafp_ryukyu.6 popup = yes }
	}

	ai_chance = {
		value = 0
	}
}
