korean_yangban_clothes_trigger = {
	exists = scope:pop
	scope:pop = {
		culture = cu:korean
     	OR = {
			is_pop_type = aristocrats
			is_pop_type = capitalists
			is_pop_type = academics
			is_pop_type = clerks
			is_pop_type = shopkeepers
			is_pop_type = engineers
			is_pop_type = bureaucrats
			is_pop_type = clergymen
     	}		
	}
}

korean_king_clothes_trigger = {
	exists = scope:character
	scope:character = {
		is_ruler = yes
		OR = {
			OWNER = c:KOR
			culture = cu:korean
		}
		owner = {
			has_law = law_type:law_monarchy
			NOR = {
				has_variable = lia_variable_western_gbr
				has_variable = lia_variable_western_fra
				has_variable = lia_variable_western_pru
			}			
		}		
	}
}

korean_soldier_clothes_trigger = {
	exists = scope:pop
	scope:pop = {
		culture = cu:korean
     	OR = {
			is_pop_type = soldiers
			is_pop_type = officers		
     	}		
	}
}

korean_general_clothes_trigger = {
	exists = scope:character
	scope:character = {
		OR = {
			OWNER = c:KOR
			culture = cu:korean
		}	
		OR = {
			has_role = general
			has_role = admiral
		}
	}
}
