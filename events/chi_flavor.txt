	type = country_event
	placement = ROOT

	desc = chi_flavor.1.d

	duration = 3

	event_image = {
		texture = "gfx/event_pictures/china/coronation.png"

				birth_date = 1835.11.29

				on_created = {
				}
			}
		}
	}

	option = {
		name = chi_flavor.1.a
		default_option = yes
		
		ai_chance = {
			base = 100
		}
	}
