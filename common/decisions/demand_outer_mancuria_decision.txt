demand_outer_mancuria_decision = {
	is_shown = {
		exists = c:CHI
		exists = c:RUS
		ROOT = c:RUS
		country_rank >= rank_value:great_power
		has_global_variable = heavenly_kingdom_start_var
		c:CHI = { country_or_subject_owns_entire_state_region = STATE_OUTER_MANCHURIA }
	}
	
	possible = {
		c:CHI = { has_war_with = c:GBR }
		NOT = {
			is_diplomatic_play_enemy_of = c:CHI
			has_truce_with = c:CHI
		}
	}
	
	when_taken = {
		s:STATE_OUTER_MANCHURIA = { add_claim = c:RUS }
		trigger_event = { id = second_opium_war.3 popup = yes }
	}

	ai_chance = {
		base = 0
		modifier = {
			trigger = {
				has_global_variable = arrow_incident
				OR = {
					game_date >= 1858
					c:CHI = {
						any_scope_war = {
							is_war_participant = c:TPG
							has_war_exhaustion = {
								target = c:CHI
								value > 20
							}
						}
					}
					c:CHI = {
						any_scope_war = {
							is_war_participant = c:GBR
							has_war_exhaustion = {
								target = c:CHI
								value > 20
							}
						}
					}
				}
			}
			add = 1
		}
		modifier = {
			trigger = {
				OR = {
					is_in_civil_war = yes
					any_country = {
						country_rank = rank_value:great_power
						is_diplomatic_play_enemy_of = ROOT
					}
				}
			}
			add = -1000
		}
	}
}
