russo_chinese.1 = {
	type = country_event

		}

			limit = {
				OR = {
					country_has_primary_culture = cu:manchu
					country_has_primary_culture = cu:han
				}
				OR = { # 추가
					any_scope_state = {
						state_region = s:STATE_OUTER_MANCHURIA

						owner = { NOT = { is_subject_type = subject_type_military_headquarters_china } }
					}
					any_subject_or_below = {
						is_subject_type = subject_type_military_headquarters_china

							state_region = s:STATE_OUTER_MANCHURIA
						}

				#	is_split_state = no
				#	is_capital = no
				#}

				is_player = no
			}
			save_scope_as = china_scope
		}

		name = russo_chinese.1.a
		default_option = yes

			limit = {

		# 			name = modifier_ruler_of_the_east
		# 			days = long_modifier_time
		# 		}
		# 	}
		# }
		every_country = { # 추가
			limit = {
				OR = {
					THIS = scope:china_scope
					is_subject_type = subject_type_military_headquarters_china
				}
			}
			every_scope_state = {
				limit = {
					state_region = s:STATE_OUTER_MANCHURIA
				}
				set_state_owner = c:RUS
				add_modifier = {
					name = modifier_ruler_of_the_east
					days = long_modifier_time
				}
			}
		}
	}

	option = {
		name = russo_chinese.1.b
		if = {
			limit = {
				scope:china_scope = {
					owes_obligation_to = c:RUS

		# 			scope:china_scope ?= owner

	duration = 3

	trigger = {

	}

	immediate = {
		s:STATE_OUTER_MANCHURIA ?= {
			random_scope_state = {
				save_scope_as = vladivostok_scope
			}
		}
	}

	option = {
		name = russo_chinese.3.a
		default_option = yes
		scope:vladivostok_scope = {
			if = {
				limit = {
					NOT = {
						has_building = building_naval_administration
					}
				}
				create_building={
	 				building="building_naval_administration"
	 				level=2
	 				reserves=1
	 			}
			}
 			add_modifier = modifier_vladivostok_navy
			add_homeland = cu:russian # 추가
			create_mass_migration = { # 추가
				origin = ROOT
				culture = cu:russian
			}
		}
		add_modifier = {
			name = modifier_vladivostok_navy_country
			days = normal_modifier_time
		}
	}

	option = {
		name = russo_chinese.3.b
		scope:vladivostok_scope = {
			if = {
				limit = {
					NOT = {
						has_building = building_port
					}
				}
				create_building={
	 				building="building_port"
	 				level=2
	 				reserves=1
	 			}
			}
 			add_modifier = modifier_vladivostok_civilian
			add_homeland = cu:russian # 추가
			create_mass_migration = { # 추가
				origin = ROOT
				culture = cu:russian
			}
 			add_loyalists = {
 				value = large_radicals
 				strata = upper
 			}
		}
	}
