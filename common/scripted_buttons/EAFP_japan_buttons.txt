	name = "japan_tutorial_button"
	desc = "japan_tutorial_button_desc"

	visible = {
		always = yes
	}

	possible = {
		root.var:japan_tutorial_var < 30
	}

	effect = {
		trigger_event = {
			id = eafp_japan.1
		}
		change_variable = {
			name = japan_tutorial_var
			add = 30
		}
		set_variable = {
            name = lia_var_regent_character
            value = root.heir
        }
	}
}
