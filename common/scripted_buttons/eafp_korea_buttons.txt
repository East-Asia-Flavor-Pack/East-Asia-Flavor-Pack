				}

				CIV = -10
			}
		}
		every_scope_state = {
			limit = {
				is_incorporated = yes

					region = sr:region_manchuria
					region = sr:region_japan
				}
			}
			add_modifier = {
				name = eafp_modi_imsul_state_lockdown

			}
		}
	}
