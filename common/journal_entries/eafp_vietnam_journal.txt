je_khmer_issue = { # 크메르 문제
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	group = je_group_historical_content

	is_shown_in_lobby = {
		c:DAI ?= THIS
	}

	immediate = {
		c:CAM = { save_scope_as = cambodia_country_scope }
		cu:khmer = {
			save_scope_as = khmer_culture_scope
		}
	}

	modifiers_while_active = {
		dai_raging_rebellion_modifier
	}
	
	complete = {
		NOT = { exists = c:CAM }
		any_scope_state = {
			filter = {
				state_region = { is_homeland = cu:khmer }
				culture_percent_state = {
					target = cu:khmer
					value > 0
				}
			}
			count = all
			state_region = { is_homeland = cu:khmer }
			is_incorporated = yes
			loyalist_fraction = {
				value >= 0.20
				culture = cu:khmer
			}
		}
		OR = {
			any_scope_state = {
				filter = {
					state_region = { is_homeland = cu:khmer }
					culture_percent_state = {
						target = cu:khmer
						value > 0
					}
				}
				count = all
				state_region = { is_homeland = cu:khmer }
				culture_percent_state = {
					target = cu:vietnamese
					value >= 0.2
				}
			}
			cultural_acceptance_base = { target = cu:khmer value >= acceptance_status_4 }
		}
	}

	on_complete = {
		trigger_event = { id = khmer_issue_events.1 }
	}

	fail = {
		# je_khmer_issue_fail
		c:CAM ?= {
			NOT = { is_subject_of = c:DAI }
		}
	}

	on_fail = {
		trigger_event = { id = khmer_issue_events.2 }
	}

	invalid = {
		hidden_trigger = {
			NAND = {
				exists = c:CAM
				any_diplomatic_play = {
					initiator = c:CAM
					target = ROOT
					is_diplomatic_play_type = dp_secession
				}
			}
		}
		NOT = {
			any_scope_state = {
				state_region = { is_homeland = cu:khmer }
			}
		}
	}

	on_monthly_pulse = {
		effect = {
			
		}
	}

	weight = 100

	should_be_pinned_by_default = yes
}
