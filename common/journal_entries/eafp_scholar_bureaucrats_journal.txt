	icon = "gfx/interface/icons/law_icons/imperial_examination.dds"

	group = je_group_internal_affairs

	possible = {
		has_law = law_type:law_scholar_bureaucrats
	}

	complete = {
		NOT = { has_law_or_variant = law_type:law_state_religion }
		NOT = { has_law_or_variant = law_type:law_peasant_levies }
		NOT = { has_law_or_variant = law_type:law_serfdom }
	}

	on_complete = {
		if = {
			limit = { c:KOR ?= THIS }
			trigger_event = eafp_korean_reformation_events.103
		}
		else = {
			activate_law = law_type:law_appointed_bureaucrats
		}
	}
	
	invalid = {
		NOT = { has_law = law_type:law_scholar_bureaucrats }
	}

	on_invalid = {
		
	}
}
